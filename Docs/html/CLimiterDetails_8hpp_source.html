65 /*!
66  * \brief Common small functions used by limiters.
67  * \ingroup FvmAlgos
68  */
69 template<class Type = su2double>
70 struct LimiterHelpers
71 {
72  FORCEINLINE static Type epsilon() {return std::numeric_limits<passivedouble>::epsilon();}
73 
74  FORCEINLINE static Type venkatFunction(const Type& proj, const Type& delta, const Type& eps2)
75  {
76  Type y = delta*(delta+proj) + eps2;
77  return (y + delta*proj) / (y + 2*proj*proj);
78  }
79 
80  FORCEINLINE static Type vanAlbadaFunction(const Type& proj, const Type& delta, const Type& eps)
81  {
82  return delta * (2*proj + delta) / (4*pow(proj, 2) + pow(delta, 2) + eps);
83  }
84 
85  FORCEINLINE static Type raisedSine(const Type& dist)
86  {
87  Type factor = 0.5*(1.0+dist+sin(PI_NUMBER*dist)/PI_NUMBER);
88  return max(0.0, min(factor, 1.0));
89  }
90 };

162 /*!
163  * \brief Venkatakrishnan-Wang specialization.
164  * \ingroup FvmAlgos
165  */
166 template<>
167 struct CLimiterDetails<LIMITER::VENKATAKRISHNAN_WANG>
168 {
169  static su2activevector sharedMin, sharedMax;
170  su2activevector eps2;
171 
172  /*!
173  * \brief Store the solution range based eps^2 parameter.
174  */
175  template<class FieldType>
176  inline void preprocess(CGeometry& geometry, const CConfig& config, size_t varBegin,
177  size_t varEnd, const FieldType& field)
178  {
179  /*--- Determine the max and min global value for each variable. ---*/
180 
181  su2double largeNum = 0.1*std::numeric_limits<passivedouble>::max();
182 
183  /*--- Allocate the static members (shared between threads) to
184  * perform the reduction across all threads in the rank. ---*/
185 
186  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
187  {
188  sharedMin.resize(varEnd) = largeNum;
189  sharedMax.resize(varEnd) =-largeNum;
190  }
191  END_SU2_OMP_SAFE_GLOBAL_ACCESS
192 
193  /*--- Per thread reduction. ---*/
194 
195  su2activevector localMin(varEnd), localMax(varEnd);
196  localMin = largeNum;
197  localMax =-largeNum;
198 
199  SU2_OMP_FOR_(schedule(static, 512) SU2_NOWAIT)
200  for(size_t iPoint = 0; iPoint < geometry.GetnPointDomain(); ++iPoint)
201  {
202  for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
203  {
204  localMin(iVar) = min(localMin(iVar), field(iPoint, iVar));
205  localMax(iVar) = max(localMax(iVar), field(iPoint, iVar));
206  }
207  }
208  END_SU2_OMP_FOR
209 
210  /*--- Per rank reduction. ---*/
211 
212  SU2_OMP_CRITICAL
213  for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
214  {
215  sharedMin(iVar) = min(sharedMin(iVar), localMin(iVar));
216  sharedMax(iVar) = max(sharedMax(iVar), localMax(iVar));
217  }
218  END_SU2_OMP_CRITICAL
219 
220  /*--- Global reduction. ---*/
221 
222  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
223  {
224  localMin = sharedMin;
225  SU2_MPI::Allreduce(localMin.data(), sharedMin.data(), varEnd, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
226 
227  localMax = sharedMax;
228  SU2_MPI::Allreduce(localMax.data(), sharedMax.data(), varEnd, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
229  }
230  END_SU2_OMP_SAFE_GLOBAL_ACCESS
231 
232  /*--- Compute eps^2 (each thread has its own copy of it). ---*/
233 
234  eps2.resize(varEnd);
235  su2double K = config.GetVenkat_LimiterCoeff();
236 
237  for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
238  {
239  su2double range = sharedMax(iVar) - sharedMin(iVar);
240  eps2(iVar) = max(pow(K*range, 2), LimiterHelpers<>::epsilon());
241  }
242  }
243 
244  /*!
245  * \brief No geometric modification for this kind of limiter.
246  */
247  template<class... Ts>
248  inline su2double geometricFactor(Ts&...) const {return 1.0;}
249 
250  /*!
251  * \brief Smooth function that disables limiting in smooth regions.
252  */
253  inline su2double limiterFunction(size_t iVar, su2double proj, su2double delta) const
254  {
255  AD::SetPreaccIn(eps2(iVar));
256  return LimiterHelpers<>::venkatFunction(proj, delta, eps2(iVar));
257  }
258 };
