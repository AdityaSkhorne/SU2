34 void COutputLegacy::ComputeTurboPerformance(CSolver *solver_container, CGeometry *geometry, CConfig *config) {
35 
36  CFluidModel *FluidModel;
37  unsigned short nDim = geometry->GetnDim();
38  unsigned short iMarkerTP, iSpan, iDim, iStage, iBlade;
39  unsigned short nMarkerTP = config->GetnMarker_Turbomachinery();
40  FluidModel = solver_container->GetFluidModel();
41  su2double area, absVel2, soundSpeed, mach, tangVel, tangVel2, *relVel, relVel2;
42  su2double relPressureIn, relPressureOut, enthalpyOutIs, relVelOutIs2;
43  relVel = new su2double[nDim];
44  su2double muLam, kine, omega, nu;
45  bool turbulent = ((config->GetKind_Solver() == MAIN_SOLVER::RANS) || (config->GetKind_Solver() == MAIN_SOLVER::DISC_ADJ_RANS));
46  bool menter_sst = (config->GetKind_Turb_Model() == TURB_MODEL::SST);
47 
48  unsigned short nBladesRow, nStages;
49 
50  nBladesRow = config->GetnMarker_Turbomachinery();
51  nStages = SU2_TYPE::Int(nBladesRow/2);
52 
53 
54  /*--- Compute BC imposed value for convergence monitoring ---*/
55  for(iMarkerTP = 0; iMarkerTP < nMarkerTP; iMarkerTP++ ){
56  for(iSpan = 0; iSpan < config->GetnSpan_iZones(iMarkerTP) + 1; iSpan++){
57  if(config->GetRampOutletPressure() && config->GetInnerIter() > 0){
58  PressureOut_BC[iMarkerTP][iSpan] = config->GetMonitorOutletPressure()/config->GetPressure_Ref();
59  }
60  FluidModel->SetTDState_PT(config->GetTotalPressureIn_BC(), config->GetTotalTemperatureIn_BC());
61  TotalEnthalpyIn_BC[iMarkerTP][iSpan] = FluidModel->GetStaticEnergy()+ FluidModel->GetPressure()/FluidModel->GetDensity();
62  EntropyIn_BC[iMarkerTP][iSpan] = FluidModel->GetEntropy();
63  }
64  }
65 
66  /*--- Compute performance for each blade ---*/
67  for(iMarkerTP = 0; iMarkerTP < nMarkerTP; iMarkerTP++ ){
68  for(iSpan = 0; iSpan < config->GetnSpan_iZones(iMarkerTP) + 1; iSpan++){

70 
71  /*--- INFLOW ---*/
72  /*--- Retrieve Inflow primitive quantities ---*/
73  DensityIn[iMarkerTP][iSpan] = solver_container->GetDensityIn(iMarkerTP, iSpan);
74  PressureIn[iMarkerTP][iSpan] = solver_container->GetPressureIn(iMarkerTP, iSpan);
75 
76  absVel2 = 0.0;
77 
78  for (iDim = 0; iDim < nDim; iDim++){
79  TurboVelocityIn[iMarkerTP][iSpan][iDim] = solver_container->GetTurboVelocityIn(iMarkerTP, iSpan)[iDim];
80  absVel2 += TurboVelocityIn[iMarkerTP][iSpan][iDim]*TurboVelocityIn[iMarkerTP][iSpan][iDim];
81  }
82  TurboVelocityIn[iMarkerTP][iSpan][nDim] = sqrt(absVel2);
83 
84  TRadius[iMarkerTP][iSpan] = geometry->GetTurboRadiusIn(iMarkerTP, iSpan);
85  area = geometry->GetSpanAreaIn(iMarkerTP, iSpan);
86 
87  /*--- Compute static Inflow quantities ---*/
88  FluidModel->SetTDState_Prho(PressureIn[iMarkerTP][iSpan], DensityIn[iMarkerTP][iSpan]);
89  EntropyIn[iMarkerTP][iSpan] = FluidModel->GetEntropy();
90  MassFlowIn[iMarkerTP][iSpan] = config->GetnBlades(iMarkerTP)*DensityIn[iMarkerTP][iSpan]*TurboVelocityIn[iMarkerTP][iSpan][0]*area;
91  AbsFlowAngleIn[iMarkerTP][iSpan] = atan(TurboVelocityIn[iMarkerTP][iSpan][1]/TurboVelocityIn[iMarkerTP][iSpan][0]);
92  EnthalpyIn[iMarkerTP][iSpan] = FluidModel->GetStaticEnergy() + PressureIn[iMarkerTP][iSpan]/DensityIn[iMarkerTP][iSpan];
93  soundSpeed = FluidModel->GetSoundSpeed();
94 
95 
96  /*--- Compute Total Inflow quantities ---*/
97  TotalEnthalpyIn[iMarkerTP][iSpan] = EnthalpyIn[iMarkerTP][iSpan] + 0.5*absVel2;
98  FluidModel->SetTDState_hs(TotalEnthalpyIn[iMarkerTP][iSpan], EntropyIn[iMarkerTP][iSpan]);
99  TotalPressureIn[iMarkerTP][iSpan] = FluidModel->GetPressure();
100  TotalTemperatureIn[iMarkerTP][iSpan] = FluidModel->GetTemperature();
101 
102  /*--- Retrieve Inflow relative quantities ---*/
103  tangVel = geometry->GetTangGridVelIn(iMarkerTP, iSpan);
104  tangVel2 = tangVel*tangVel;
105 
106  for (iDim = 0; iDim < nDim; iDim++){
107  relVel[iDim] = TurboVelocityIn[iMarkerTP][iSpan][iDim];
108  }
109  relVel[1] -= tangVel;
110 
111  relVel2 = 0.0;
112  for (iDim = 0; iDim < nDim; iDim++){
113  relVel2 += relVel[iDim]*relVel[iDim];
114  }
115 
116  /*--- Compute Total relative Inflow quantities ---*/
117  RothalpyIn[iMarkerTP][iSpan] = EnthalpyIn[iMarkerTP][iSpan] + 0.5*relVel2 - 0.5*tangVel2;
118  FluidModel->SetTDState_hs(RothalpyIn[iMarkerTP][iSpan], EntropyIn[iMarkerTP][iSpan]);
119  relPressureIn = FluidModel->GetPressure();
120 
121  /*--- Compute kinematic relative Inflow quantities ---*/
122  FlowAngleIn[iMarkerTP][iSpan] = atan(relVel[1]/relVel[0]);
123  mach = 0.0;
124  for (iDim = 0; iDim < nDim; iDim++){
125  MachIn[iMarkerTP][iSpan][iDim] = relVel[iDim]/soundSpeed;
126  mach = MachIn[iMarkerTP][iSpan][iDim]*MachIn[iMarkerTP][iSpan][iDim];
127  }
128  MachIn[iMarkerTP][iSpan][nDim] = sqrt(mach);
129 
130  /*--- Compute Turbulent Inflow quantities ---*/
131  if(turbulent){
132  FluidModel->SetTDState_Prho(PressureIn[iMarkerTP][iSpan], DensityIn[iMarkerTP][iSpan]);
133  muLam = FluidModel->GetLaminarViscosity();
134  if(menter_sst){
135  kine = solver_container->GetKineIn(iMarkerTP, iSpan);
136  omega = solver_container->GetOmegaIn(iMarkerTP, iSpan);
137  TurbIntensityIn[iMarkerTP][iSpan] = sqrt(2.0/3.0*kine/absVel2);
138  Turb2LamViscRatioIn[iMarkerTP][iSpan] = DensityIn[iMarkerTP][iSpan]*kine/(muLam*omega);
139 // TurbIntensityIn[iMarkerTP][iSpan] = kine;
140 // Turb2LamViscRatioIn[iMarkerTP][iSpan] = omega;
141  }
142  else{
143  nu = solver_container->GetNuIn(iMarkerTP, iSpan);
144  NuFactorIn[iMarkerTP][iSpan] = nu*DensityIn[iMarkerTP][iSpan]/muLam;
145  if (config->GetSAParsedOptions().bc) {
146  NuFactorIn[iMarkerTP][iSpan] = nu*DensityIn[iMarkerTP][iSpan]/muLam/0.005;
147  }
148  }
149  }
150 
151  /*--- OUTFLOW ---*/
152  /*--- Retrieve Outflow primitive quantities ---*/
153  DensityOut[iMarkerTP][iSpan] = solver_container->GetDensityOut(iMarkerTP, iSpan);
154  PressureOut[iMarkerTP][iSpan] = solver_container->GetPressureOut(iMarkerTP, iSpan);
155  absVel2 = 0.0;
156 
157  for (iDim = 0; iDim < nDim; iDim++){
158  TurboVelocityOut[iMarkerTP][iSpan][iDim] = solver_container->GetTurboVelocityOut(iMarkerTP, iSpan)[iDim];
159  absVel2 += TurboVelocityOut[iMarkerTP][iSpan][iDim]*TurboVelocityOut[iMarkerTP][iSpan][iDim];
160  }
161  TurboVelocityOut[iMarkerTP][iSpan][nDim] = sqrt(absVel2);
162 
163 
164  for (iDim = 0; iDim < 3; iDim++){
165  }
166  area = geometry->GetSpanAreaOut(iMarkerTP, iSpan);
167 
168 
169  /*--- Compute all the Outflow quantities ---*/
170  FluidModel->SetTDState_Prho(PressureOut[iMarkerTP][iSpan], DensityOut[iMarkerTP][iSpan]);
171  EntropyOut[iMarkerTP][iSpan] = FluidModel->GetEntropy();
172  MassFlowOut[iMarkerTP][iSpan] = config->GetnBlades(iMarkerTP)*DensityOut[iMarkerTP][iSpan]*TurboVelocityOut[iMarkerTP][iSpan][0]*area;
173  AbsFlowAngleOut[iMarkerTP][iSpan] = atan(TurboVelocityOut[iMarkerTP][iSpan][1]/TurboVelocityOut[iMarkerTP][iSpan][0]);
174  EnthalpyOut[iMarkerTP][iSpan] = FluidModel->GetStaticEnergy() + PressureOut[iMarkerTP][iSpan]/DensityOut[iMarkerTP][iSpan];
175  soundSpeed = FluidModel->GetSoundSpeed();
176 
177  /*--- Compute Total Outflow quantities ---*/
178  TotalEnthalpyOut[iMarkerTP][iSpan] = EnthalpyOut[iMarkerTP][iSpan] + 0.5*absVel2;
179  FluidModel->SetTDState_hs(TotalEnthalpyOut[iMarkerTP][iSpan], EntropyOut[iMarkerTP][iSpan]);
180  TotalPressureOut[iMarkerTP][iSpan] = FluidModel->GetPressure();
181  TotalTemperatureOut[iMarkerTP][iSpan] = FluidModel->GetTemperature();
182 
183  /*--- Retrieve relative Outflow quantities ---*/
184  tangVel = geometry->GetTangGridVelOut(iMarkerTP, iSpan);
185  tangVel2 = tangVel*tangVel;
186 
187  for (iDim = 0; iDim < nDim; iDim++){
188  relVel[iDim] = TurboVelocityOut[iMarkerTP][iSpan][iDim];
189  }
190  relVel[1] -= tangVel;
191 
192  relVel2 = 0.0;
193  for (iDim = 0; iDim < nDim; iDim++){
194  relVel2 += relVel[iDim]*relVel[iDim];
195  }
196 
197  /*--- Compute Total relative Outflow quantities ---*/
198  RothalpyOut[iMarkerTP][iSpan] = EnthalpyOut[iMarkerTP][iSpan] + 0.5*relVel2 - 0.5*tangVel2;
199  FluidModel->SetTDState_hs(RothalpyOut[iMarkerTP][iSpan], EntropyOut[iMarkerTP][iSpan]);
200  relPressureOut = FluidModel->GetPressure();
201 
202  /*--- Compute isentropic Outflow quantities ---*/
203  FluidModel->SetTDState_Ps(PressureOut[iMarkerTP][iSpan], EntropyIn[iMarkerTP][iSpan]);
204  enthalpyOutIs = FluidModel->GetStaticEnergy() + PressureOut[iMarkerTP][iSpan]/FluidModel->GetDensity();
205  relVelOutIs2 = 2*(RothalpyOut[iMarkerTP][iSpan] - enthalpyOutIs) + tangVel2;
206 
207 
208  /*--- Compute kinematic relative Outflow quantities ---*/
209  FlowAngleOut[iMarkerTP][iSpan] = atan(relVel[1]/relVel[0]);
210  mach = 0.0;
211  for (iDim = 0; iDim < nDim; iDim++){
212  MachOut[iMarkerTP][iSpan][iDim] = relVel[iDim]/soundSpeed;
213  mach = MachOut[iMarkerTP][iSpan][iDim]*MachOut[iMarkerTP][iSpan][iDim];
214  }
215  MachOut[iMarkerTP][iSpan][nDim] = sqrt(mach);
216 
217  /*--- Compute Turbulent Outflow quantities ---*/
218  if(turbulent){
219  FluidModel->SetTDState_Prho(PressureOut[iMarkerTP][iSpan], DensityOut[iMarkerTP][iSpan]);
220  muLam = FluidModel->GetLaminarViscosity();
221  if(menter_sst){
222  kine = solver_container->GetKineOut(iMarkerTP, iSpan);
223  omega = solver_container->GetOmegaOut(iMarkerTP, iSpan);
224  TurbIntensityOut[iMarkerTP][iSpan] = sqrt(2.0/3.0*kine/absVel2);
225  Turb2LamViscRatioOut[iMarkerTP][iSpan] = DensityOut[iMarkerTP][iSpan]*kine/(muLam*omega);
226 // TurbIntensityOut[iMarkerTP][iSpan] = kine;
227 // Turb2LamViscRatioOut[iMarkerTP][iSpan] = omega;
228  }
229  else{
230  nu = solver_container->GetNuOut(iMarkerTP, iSpan);
231  NuFactorOut[iMarkerTP][iSpan] = nu*DensityOut[iMarkerTP][iSpan]/muLam;
232  if (config->GetSAParsedOptions().bc) {
233  NuFactorOut[iMarkerTP][iSpan] = nu*DensityOut[iMarkerTP][iSpan]/muLam/0.005;
234  }
235  }
236  }
237 
238  /*--- TURBO-PERFORMANCE---*/
239  EntropyGen[iMarkerTP][iSpan] = (EntropyOut[iMarkerTP][iSpan] - EntropyIn[iMarkerTP][iSpan])/abs(EntropyIn_BC[iMarkerTP][iSpan] + 1);
240  EulerianWork[iMarkerTP][iSpan] = TotalEnthalpyIn[iMarkerTP][iSpan] - TotalEnthalpyOut[iMarkerTP][iSpan];
241  TotalPressureLoss[iMarkerTP][iSpan] = (relPressureIn - relPressureOut)/(relPressureIn - PressureOut[iMarkerTP][iSpan]);
242  KineticEnergyLoss[iMarkerTP][iSpan] = 2*(EnthalpyOut[iMarkerTP][iSpan] - enthalpyOutIs)/relVelOutIs2;
243  PressureRatio[iMarkerTP][iSpan] = TotalPressureOut[iMarkerTP][iSpan]/TotalPressureIn[iMarkerTP][iSpan];
244  EnthalpyOutIs[iMarkerTP][iSpan] = (pow(TotalPressureOut[iMarkerTP][iSpan]/TotalPressureIn[iMarkerTP][iSpan], 0.4/1.4) - 1.0)/(TotalTemperatureOut[iMarkerTP][iSpan]/TotalTemperatureIn[iMarkerTP][iSpan] -1.0);
245  }
246  }
247 
248  if(nBladesRow > 1){
249  /*--- Compute performance for each stage ---*/
250 
251  EulerianWork[nBladesRow + nStages][nSpanWiseSections] = 0.0;
252  /*---Comnpute performance for each stage---*/
253  for(iStage = 0; iStage < nStages; iStage++ ){
254  FluidModel->SetTDState_Ps(PressureOut[iStage*2 +1][config->GetnSpan_iZones(iStage*2 +1)], EntropyIn[iStage*2][config->GetnSpan_iZones(iStage*2)]);
255  EnthalpyOutIs[nBladesRow + iStage][nSpanWiseSections] = FluidModel->GetStaticEnergy() + PressureOut[iStage*2 +1][config->GetnSpan_iZones(iStage*2 +1)]/FluidModel->GetDensity();
256  FluidModel->SetTDState_Prho(PressureOut[iStage*2 +1][config->GetnSpan_iZones(iStage*2 +1)], DensityOut[iStage*2 +1][config->GetnSpan_iZones(iStage*2 +1)]);
257  absVel2 = 0.0;
258  for (iDim = 0; iDim<nDim; iDim++)
259  absVel2 += TurboVelocityOut[iStage*2 +1][config->GetnSpan_iZones(iStage*2 +1)][iDim]*TurboVelocityOut[iStage*2 +1][config->GetnSpan_iZones(iStage*2 +1)][iDim];
260  TotalEnthalpyOutIs[nBladesRow + iStage][nSpanWiseSections] = EnthalpyOutIs[nBladesRow + iStage][nSpanWiseSections] + 0.5*absVel2;
261 
262  TotalTotalEfficiency[nBladesRow + iStage][nSpanWiseSections] = (TotalEnthalpyIn[iStage*2][config->GetnSpan_iZones(iStage*2)] - TotalEnthalpyOut[iStage*2 + 1][config->GetnSpan_iZones(iStage*2 +1)]);
263  TotalTotalEfficiency[nBladesRow + iStage][nSpanWiseSections] /= (TotalEnthalpyIn[iStage*2][config->GetnSpan_iZones(iStage*2)] - TotalEnthalpyOutIs[nBladesRow + iStage][nSpanWiseSections]);
264  TotalStaticEfficiency[nBladesRow + iStage][nSpanWiseSections] = (TotalEnthalpyIn[iStage*2][config->GetnSpan_iZones(iStage*2)] - TotalEnthalpyOut[iStage*2 + 1][config->GetnSpan_iZones(iStage*2+1)]);
265  TotalStaticEfficiency[nBladesRow + iStage][nSpanWiseSections] /= (TotalEnthalpyIn[iStage*2][config->GetnSpan_iZones(iStage*2)] - EnthalpyOutIs[nBladesRow + iStage][nSpanWiseSections]);
266  PressureRatio[nBladesRow + iStage][nSpanWiseSections] = (PressureRatio[iStage*2][config->GetnSpan_iZones(iStage*2)]*PressureOut[iStage*2][config->GetnSpan_iZones(iStage*2)]/PressureOut[iStage*2 + 1][config->GetnSpan_iZones(iStage*2+1)]);
267  MassFlowIn[nBladesRow + iStage][nSpanWiseSections] = MassFlowIn[iStage*2][config->GetnSpan_iZones(iStage*2)];
268  MassFlowOut[nBladesRow + iStage][nSpanWiseSections] = MassFlowOut[iStage*2 + 1][config->GetnSpan_iZones(iStage*2+1)];
269  EntropyGen[nBladesRow + iStage][nSpanWiseSections] = EntropyGen[iStage*2 + 1][config->GetnSpan_iZones(iStage*2 +1)] + EntropyGen[iStage*2][config->GetnSpan_iZones(iStage*2)];
270 
271  }
272 
273  /*---Compute turbo performance for full machine---*/
274  FluidModel->SetTDState_Ps(PressureOut[nBladesRow-1][config->GetnSpan_iZones(nBladesRow-1)], EntropyIn[0][config->GetnSpan_iZones(0)]);

277  absVel2 = 0.0;

280

282  TotalTotalEfficiency[nBladesRow + nStages][nSpanWiseSections] /= (TotalEnthalpyIn[0][config->GetnSpan_iZones(0)] - TotalEnthalpyOutIs[nBladesRow + nStages][nSpanWiseSections]);

292  }
293  }
