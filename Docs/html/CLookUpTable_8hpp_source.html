90  /*! \brief
91  * returns the index to the variable in the lookup table.
92  */
93  inline unsigned int GetIndexOfVar(const std::string& nameVar) const {
94  int index = find(names_var.begin(), names_var.end(), nameVar) - names_var.begin();
95  if (index == int(names_var.size())) {
96  index = -1;
97  std::string error_msg = "Variable '";
98  error_msg.append(nameVar);
99  error_msg.append("' is not in the lookup table.");
100  SU2_MPI::Error(error_msg, CURRENT_FUNCTION);
101  }
102  return index;
103  }

192  /*!
193  * \brief compute the area of a triangle given the 3 points of the triangle.
194  * \param[in] x1 - the coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
195  * \param[in] y1 - the coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
196  * \param[in] x2 - the coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
197  * \param[in] y2 - the coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
198  * \param[in] x3 - the coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
199  * \param[in] y3 - the coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
200  * \returns the absolute value of the area of the triangle.
201  */
202  inline su2double TriArea(su2double x1, su2double y1, su2double x2, su2double y2, su2double x3, su2double y3) {
203  return abs((x1 * (y2 - y3) + x2 * (y3 - y1) + x3 * (y1 - y2)) * 0.5);
204  }
