33 /*!
34  * \brief Symbols that need to be defined before including this header:
35  * \param[in] ARRAY_T - The desired specialization of simd::Array.
36  * \param[in] SCALAR_T - Scalar type.
37  * \param[in] REGISTER_T - Intrinsic type.
38  * \param[in] SIZE_TAG - Dummy object associated with the simd size.
39  * \param[in] ARRAY_BOILERPLATE - Generates the general ctors, access, etc.
40  * \note On top of that, the intrinsic functions must be wrapped to
41  * strip their type / size characteristics, e.g. _mm256_add_pd -> add_p,
42  * overload resolution will do the rest. The first four symbols are
43  * undefined once we are done using them.
44  */
45 template<>
46 class ARRAY_T {
47 #define FOREACH SU2_OMP_SIMD for(size_t k=0; k<Size; ++k)
48  template<class F, class S>
49  FORCEINLINE static S second(F, S s) { return s; }
50 public:
51  using Scalar = SCALAR_T;
52  using Register = REGISTER_T;
53  enum : size_t {Align = alignof(Register)};
54  enum : size_t {Size = sizeof(Register) / sizeof(Scalar)};
55 
56  /*--- The infamous union "hack", sue me. ---*/
57  union {
58  Register reg;
59  Scalar x_[Size];
60  };
61 
62  /*--- Same basic construction operations as the general case. ---*/
63 
64  ARRAY_BOILERPLATE
65 
66  /*--- Special construction using the "register type" directly. ---*/
67 
68  FORCEINLINE Array(Register y) { reg = y; }
69  FORCEINLINE Array(const Array& other) noexcept { reg = other.reg; }
70 
71  /*--- Specialized construction primitives. ---*/
72 
73  FORCEINLINE void bcast(Scalar x) { reg = set1_p(SIZE_TAG, x); }
74  FORCEINLINE void load(const Scalar* ptr) { reg = loadu_p(SIZE_TAG, ptr); }
75  FORCEINLINE void loada(const Scalar* ptr) { reg = load_p(SIZE_TAG, ptr); }
76  FORCEINLINE void store(Scalar* ptr) const { storeu_p(ptr, reg); }
77  FORCEINLINE void storea(Scalar* ptr) const { store_p(ptr, reg); }
78  FORCEINLINE void stream(Scalar* ptr) const { stream_p(ptr, reg); }
79  template<class T>
80  FORCEINLINE void gather(const Scalar* begin, const T& offsets) { FOREACH x_[k] = begin[offsets[k]]; }
81 
82  /*--- Compound assignement operators. ---*/
83 
84 #define MAKE_COMPOUND(OP,IMPL)\
85  FORCEINLINE Array& operator OP (Scalar x) { reg = IMPL(reg, set1_p(SIZE_TAG, x)); return *this; }\
86  FORCEINLINE Array& operator OP (const Array& other) noexcept { reg = IMPL(reg, other.reg); return *this; }
87  MAKE_COMPOUND(=, second)
88  MAKE_COMPOUND(+=, add_p)
89  MAKE_COMPOUND(-=, sub_p)
90  MAKE_COMPOUND(*=, mul_p)
91  MAKE_COMPOUND(/=, div_p)
92 #undef MAKE_COMPOUND
93 
94 #undef FOREACH
95 };
