31 void CDiscAdjFluidIteration::Preprocess(COutput* output, CIntegration**** integration, CGeometry**** geometry,
32  CSolver***** solver, CNumerics****** numerics, CConfig** config,
33  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
34  CFreeFormDefBox*** FFDBox, unsigned short iZone, unsigned short iInst) {
35  StartTime = SU2_MPI::Wtime();
36 
37  const auto TimeIter = config[iZone]->GetTimeIter();
38  const bool dual_time_1st = (config[iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST);
39  const bool dual_time_2nd = (config[iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
40  const bool dual_time = (dual_time_1st || dual_time_2nd);
41  const bool grid_IsMoving = config[iZone]->GetGrid_Movement();
42  const bool species = config[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE;
43  const bool heat = config[iZone]->GetWeakly_Coupled_Heat();
44 
45  auto solvers0 = solver[iZone][iInst][MESH_0];
46  auto geometries = geometry[iZone][iInst];
47 
48  /*--- For the unsteady adjoint, load direct solutions from restart files. ---*/
49 
50  if (config[iZone]->GetTime_Marching() != TIME_MARCHING::STEADY) {
51  const int Direct_Iter = static_cast<int>(config[iZone]->GetUnst_AdjointIter()) - static_cast<int>(TimeIter) - 2 + dual_time;
52 
53  /*--- For dual-time stepping we want to load the already converged solution at timestep n ---*/
54 
55  if (TimeIter == 0) {
56  if (dual_time_2nd) {
57  /*--- Load solution at timestep n-2 ---*/
58  LoadUnsteady_Solution(geometry, solver, config, iZone, iInst, Direct_Iter - 2);
59 
60  /*--- Push solution back to correct array ---*/
61 
62  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
63  auto solvers = solver[iZone][iInst][iMesh];
64 
65  solvers[FLOW_SOL]->GetNodes()->Set_Solution_time_n();
66  solvers[FLOW_SOL]->GetNodes()->Set_Solution_time_n1();
67  if (turbulent) {
68  solvers[TURB_SOL]->GetNodes()->Set_Solution_time_n();
69  solvers[TURB_SOL]->GetNodes()->Set_Solution_time_n1();
70  }
71  if (species) {
72  solvers[SPECIES_SOL]->GetNodes()->Set_Solution_time_n();
73  solvers[SPECIES_SOL]->GetNodes()->Set_Solution_time_n1();
74  }
75  if (heat) {
76  solvers[HEAT_SOL]->GetNodes()->Set_Solution_time_n();
77  solvers[HEAT_SOL]->GetNodes()->Set_Solution_time_n1();
78  }
79  if (grid_IsMoving) {
80  geometries[iMesh]->nodes->SetCoord_n();
81  geometries[iMesh]->nodes->SetCoord_n1();
82  }
83  if (config[iZone]->GetDynamic_Grid()) {
84  geometries[iMesh]->nodes->SetVolume_n();
85  geometries[iMesh]->nodes->SetVolume_nM1();
86  }
87  }
88  }
89  if (dual_time) {
90  /*--- Load solution at timestep n-1 ---*/
91  LoadUnsteady_Solution(geometry, solver, config, iZone, iInst, Direct_Iter - 1);
92 
93  /*--- Push solution back to correct array ---*/
94 
95  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
96  auto solvers = solver[iZone][iInst][iMesh];
97 
98  solvers[FLOW_SOL]->GetNodes()->Set_Solution_time_n();
99  if (turbulent) {
100  solvers[TURB_SOL]->GetNodes()->Set_Solution_time_n();
101  }
102  if (species) {
103  solvers[SPECIES_SOL]->GetNodes()->Set_Solution_time_n();
104  }
105  if (heat) {
106  solvers[HEAT_SOL]->GetNodes()->Set_Solution_time_n();
107  }
108  if (grid_IsMoving) {
109  geometries[iMesh]->nodes->SetCoord_n();
110  }
111  if (config[iZone]->GetDynamic_Grid()) {
112  geometries[iMesh]->nodes->SetVolume_n();
113  }
114  }
115  }
116 
117  /*--- Load solution timestep n ---*/
118 
119  LoadUnsteady_Solution(geometry, solver, config, iZone, iInst, Direct_Iter);
120 
121  if (config[iZone]->GetDeform_Mesh()) {
122  solvers0[MESH_SOL]->LoadRestart(geometries, solver[iZone][iInst], config[iZone], Direct_Iter, true);
123  }
124 
125  }
126  else if ((TimeIter > 0) && dual_time) {
127  /*---
128  Here the primal solutions (only working variables) are loaded and put in the correct order
129  into containers. For ALE the mesh coordinates have to be put into the
130  correct containers as well, i.e. follow the same logic for the solution.
131  Afterwards the GridVelocity is computed based on the Coordinates.
132  ---*/
133 
134  /*--- Temporarily store the loaded volumes into old containers ---*/
135  if (config[iZone]->GetDynamic_Grid()) {
136  for (auto iMesh=0; iMesh<=config[iZone]->GetnMGLevels();iMesh++) {
137  geometries[iMesh]->nodes->SetVolume_Old();
138  geometries[iMesh]->nodes->SetVolume_n_Old();
139  geometries[iMesh]->nodes->SetVolume_nM1_Old();
140  }
141  }
142 
143  /*-- Load mesh solver ---*/
144  if (config[iZone]->GetDeform_Mesh()) {
145  solvers0[MESH_SOL]->LoadRestart(geometries, solver[iZone][iInst], config[iZone], Direct_Iter, true);
146  }
147 
148  /*--- Load solution timestep n-1 | n-2 for DualTimestepping 1st | 2nd order ---*/
149  if (dual_time_1st) {
150  LoadUnsteady_Solution(geometry, solver, config, iZone, iInst, Direct_Iter - 1);
151  } else {
152  LoadUnsteady_Solution(geometry, solver, config, iZone, iInst, Direct_Iter - 2);
153 
154  /*--- Set volumes into correct containers ---*/
155  if (config[iZone]->GetDynamic_Grid()) {
156  for (auto iMesh=0; iMesh<=config[iZone]->GetnMGLevels();iMesh++) {
157  /*--- If negative iteration number, set default ---*/
158  if (Direct_Iter - 2 < 0) {
159  for(auto iPoint=0ul; iPoint<geometries[iMesh]->GetnPoint();iPoint++) {
160  geometries[iMesh]->nodes->SetVolume(iPoint,0.0);
161  }
162  }
163 
164  /*--- Set currently loaded volume to Volume_nM1 ---*/
165  geometries[iMesh]->nodes->SetVolume_n();
166  geometries[iMesh]->nodes->SetVolume_nM1();
167 
168  /*--- Set Volume_n and Volume from old containers ---*/
169  geometries[iMesh]->nodes->SetVolume_n_from_OldnM1();
170  geometries[iMesh]->nodes->SetVolume_from_Oldn();
171  }
172  }
173  }
174 
175  /*--- Temporarily store the loaded solution in the Solution_Old array ---*/
176 
177  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
178  auto solvers = solver[iZone][iInst][iMesh];
179 
180  solvers[FLOW_SOL]->Set_OldSolution();
181  if (turbulent) {
182  solvers[TURB_SOL]->Set_OldSolution();
183  }
184  if (species) {
185  solvers[SPECIES_SOL]->Set_OldSolution();
186  }
187  if (heat) {
188  solvers[HEAT_SOL]->Set_OldSolution();
189  }
190  if (grid_IsMoving) {
191  geometries[iMesh]->nodes->SetCoord_Old();
192  }
193  }
194 
195  /*--- Set Solution at timestep n to solution at n-1 ---*/
196 
197  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
198  auto solvers = solver[iZone][iInst][iMesh];
199 
200  for (auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++) {
201  solvers[FLOW_SOL]->GetNodes()->SetSolution(iPoint, solvers[FLOW_SOL]->GetNodes()->GetSolution_time_n(iPoint));
202 
203  if (grid_IsMoving) {
204  geometries[iMesh]->nodes->SetCoord(iPoint, geometries[iMesh]->nodes->GetCoord_n(iPoint));
205  }
206  if (turbulent) {
207  solvers[TURB_SOL]->GetNodes()->SetSolution(iPoint, solvers[TURB_SOL]->GetNodes()->GetSolution_time_n(iPoint));
208  }
209  if (species) {
210  solvers[SPECIES_SOL]->GetNodes()->SetSolution(iPoint, solvers[SPECIES_SOL]->GetNodes()->GetSolution_time_n(iPoint));
211  }
212  if (heat) {
213  solvers[HEAT_SOL]->GetNodes()->SetSolution(iPoint, solvers[HEAT_SOL]->GetNodes()->GetSolution_time_n(iPoint));
214  }
215  }
216  }
217  if (dual_time_1st) {
218  /*--- Set Solution at timestep n-1 to the previously loaded solution ---*/
219  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
220  auto solvers = solver[iZone][iInst][iMesh];
221 
222  for (auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++) {
223  solvers[FLOW_SOL]->GetNodes()->Set_Solution_time_n(
224  iPoint, solvers[FLOW_SOL]->GetNodes()->GetSolution_Old(iPoint));
225 
226  if (grid_IsMoving) {
227  geometries[iMesh]->nodes->SetCoord_n(iPoint, geometries[iMesh]->nodes->GetCoord_Old(iPoint));
228  }
229  if (turbulent) {
230  solvers[TURB_SOL]->GetNodes()->Set_Solution_time_n(
231  iPoint, solvers[TURB_SOL]->GetNodes()->GetSolution_Old(iPoint));
232  }
233  if (species) {
234  solvers[SPECIES_SOL]->GetNodes()->Set_Solution_time_n(
235  iPoint, solvers[SPECIES_SOL]->GetNodes()->GetSolution_Old(iPoint));
236  }
237  if (heat) {
238  solvers[HEAT_SOL]->GetNodes()->Set_Solution_time_n(
239  iPoint, solvers[HEAT_SOL]->GetNodes()->GetSolution_Old(iPoint));
240  }
241  }
242  }
243  }
244  if (dual_time_2nd) {
245  /*--- Set Solution at timestep n-1 to solution at n-2 ---*/
246  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
247  auto solvers = solver[iZone][iInst][iMesh];
248 
249  for (auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++) {
250  solvers[FLOW_SOL]->GetNodes()->Set_Solution_time_n(
251  iPoint, solvers[FLOW_SOL]->GetNodes()->GetSolution_time_n1(iPoint));
252 
253  if (grid_IsMoving) {
254  geometries[iMesh]->nodes->SetCoord_n(iPoint, geometries[iMesh]->nodes->GetCoord_n1(iPoint));
255  }
256  if (turbulent) {
257  solvers[TURB_SOL]->GetNodes()->Set_Solution_time_n(
258  iPoint, solvers[TURB_SOL]->GetNodes()->GetSolution_time_n1(iPoint));
259  }
260  if (species) {
261  solvers[SPECIES_SOL]->GetNodes()->Set_Solution_time_n(
262  iPoint, solvers[SPECIES_SOL]->GetNodes()->GetSolution_time_n1(iPoint));
263  }
264  if (heat) {
265  solvers[HEAT_SOL]->GetNodes()->Set_Solution_time_n(
266  iPoint, solvers[HEAT_SOL]->GetNodes()->GetSolution_time_n1(iPoint));
267  }
268  }
269  }
270  /*--- Set Solution at timestep n-2 to the previously loaded solution ---*/
271  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
272  auto solvers = solver[iZone][iInst][iMesh];
273 
274  for (auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++) {
275  solvers[FLOW_SOL]->GetNodes()->Set_Solution_time_n1(
276  iPoint, solvers[FLOW_SOL]->GetNodes()->GetSolution_Old(iPoint));
277 
278  if (grid_IsMoving) {
279  geometries[iMesh]->nodes->SetCoord_n1(iPoint, geometries[iMesh]->nodes->GetCoord_Old(iPoint));
280  }
281  if (turbulent) {
282  solvers[TURB_SOL]->GetNodes()->Set_Solution_time_n1(
283  iPoint, solvers[TURB_SOL]->GetNodes()->GetSolution_Old(iPoint));
284  }
285  if (species) {
286  solvers[SPECIES_SOL]->GetNodes()->Set_Solution_time_n1(
287  iPoint, solvers[SPECIES_SOL]->GetNodes()->GetSolution_Old(iPoint));
288  }
289  if (heat) {
290  solvers[HEAT_SOL]->GetNodes()->Set_Solution_time_n1(
291  iPoint, solvers[HEAT_SOL]->GetNodes()->GetSolution_Old(iPoint));
292  }
293  }
294  }
295  }
296 
297  } // else if TimeIter > 0
298 
299  /*--- Compute & set Grid Velocity via finite differences of the Coordinates. ---*/
300  if (grid_IsMoving)
301  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++)
302  geometries[iMesh]->SetGridVelocity(config[iZone]);
303 
304  } // if unsteady
305 
306  SU2_OMP_PARALLEL_(if(solvers0[ADJFLOW_SOL]->GetHasHybridParallel())) {
307 
308  /*--- Store flow solution also in the adjoint solver in order to be able to reset it later ---*/
309 
310  if (TimeIter == 0 || dual_time) {
311  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
312  auto solvers = solver[iZone][iInst][iMesh];
313  SU2_OMP_FOR_STAT(1024)
314  for (auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++)
315  solvers[ADJFLOW_SOL]->GetNodes()->SetSolution_Direct(iPoint, solvers[FLOW_SOL]->GetNodes()->GetSolution(iPoint));
316  END_SU2_OMP_FOR
317  }
318  if (turbulent && !config[iZone]->GetFrozen_Visc_Disc()) {
319  SU2_OMP_FOR_STAT(1024)
320  for (auto iPoint = 0ul; iPoint < geometries[MESH_0]->GetnPoint(); iPoint++)
321  solvers0[ADJTURB_SOL]->GetNodes()->SetSolution_Direct(iPoint, solvers0[TURB_SOL]->GetNodes()->GetSolution(iPoint));
322  END_SU2_OMP_FOR
323  }
324  if (species) {
325  SU2_OMP_FOR_STAT(1024)
326  for (auto iPoint = 0ul; iPoint < geometries[MESH_0]->GetnPoint(); iPoint++)
327  solvers0[ADJSPECIES_SOL]->GetNodes()->SetSolution_Direct(iPoint, solvers0[SPECIES_SOL]->GetNodes()->GetSolution(iPoint));
328  END_SU2_OMP_FOR
329  }
330  if (heat) {
331  SU2_OMP_FOR_STAT(1024)
332  for (auto iPoint = 0ul; iPoint < geometries[MESH_0]->GetnPoint(); iPoint++)
333  solvers0[ADJHEAT_SOL]->GetNodes()->SetSolution_Direct(iPoint, solvers0[HEAT_SOL]->GetNodes()->GetSolution(iPoint));
334  END_SU2_OMP_FOR
335  }
336  if (config[iZone]->AddRadiation()) {
337  SU2_OMP_FOR_STAT(1024)
338  for (auto iPoint = 0ul; iPoint < geometries[MESH_0]->GetnPoint(); iPoint++)
339  solvers0[ADJRAD_SOL]->GetNodes()->SetSolution_Direct(iPoint, solvers0[RAD_SOL]->GetNodes()->GetSolution(iPoint));
340  END_SU2_OMP_FOR
341  }
342  }
343 
344  solvers0[ADJFLOW_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
345  MESH_0, 0, RUNTIME_ADJFLOW_SYS, false);
346 
347  if (turbulent && !config[iZone]->GetFrozen_Visc_Disc()) {
348  solvers0[ADJTURB_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
349  MESH_0, 0, RUNTIME_ADJTURB_SYS, false);
350  }
351  if (species) {
352  solvers0[ADJSPECIES_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
353  MESH_0, 0, RUNTIME_ADJSPECIES_SYS, false);
354  }
355  if (heat) {
356  solvers0[ADJHEAT_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
357  MESH_0, 0, RUNTIME_ADJHEAT_SYS, false);
358  }
359  if (config[iZone]->AddRadiation()) {
360  solvers0[ADJRAD_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
361  MESH_0, 0, RUNTIME_ADJRAD_SYS, false);
362  }
363 
364  }
365  END_SU2_OMP_PARALLEL

368 void CDiscAdjFluidIteration::LoadUnsteady_Solution(CGeometry**** geometry, CSolver***** solver, CConfig** config,
369  unsigned short iZone, unsigned short iInst, int DirectIter) {
370 
371  auto solvers = solver[iZone][iInst];
372  const bool species = config[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE;
373 
374  if (DirectIter >= 0) {
375  if (rank == MASTER_NODE)
376  cout << " Loading flow solution from direct iteration " << DirectIter << " for zone " << iZone << "." << endl;
377 
378  solvers[MESH_0][FLOW_SOL]->LoadRestart(geometry[iZone][iInst], solvers, config[iZone], DirectIter, true);
379 
380  if (turbulent) {
381  solvers[MESH_0][TURB_SOL]->LoadRestart(geometry[iZone][iInst], solvers, config[iZone], DirectIter, false);
382  }
383  if (species) {
384  solvers[MESH_0][SPECIES_SOL]->LoadRestart(geometry[iZone][iInst], solvers, config[iZone], DirectIter, false);
385  }
386  if (config[iZone]->GetWeakly_Coupled_Heat()) {
387  solvers[MESH_0][HEAT_SOL]->LoadRestart(geometry[iZone][iInst], solvers, config[iZone], DirectIter, false);
388  }
389  } else {
390  /*--- If there is no solution file we set the freestream condition ---*/
391  if (rank == MASTER_NODE)
392  cout << " Setting freestream conditions at direct iteration " << DirectIter << " for zone " << iZone << "." << endl;
393 
394  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
395  solvers[iMesh][FLOW_SOL]->SetFreeStream_Solution(config[iZone]);
396  solvers[iMesh][FLOW_SOL]->Preprocessing(geometry[iZone][iInst][iMesh], solvers[iMesh], config[iZone], iMesh,
397  DirectIter, RUNTIME_FLOW_SYS, false);
398  if (turbulent) {
399  solvers[iMesh][TURB_SOL]->SetFreeStream_Solution(config[iZone]);
400  solvers[iMesh][TURB_SOL]->Postprocessing(geometry[iZone][iInst][iMesh], solvers[iMesh], config[iZone], iMesh);
401  }
402  if (species) {
403  solvers[iMesh][SPECIES_SOL]->SetFreeStream_Solution(config[iZone]);
404  solvers[iMesh][SPECIES_SOL]->Postprocessing(geometry[iZone][iInst][iMesh], solvers[iMesh], config[iZone], iMesh);
405  }
406  if (config[iZone]->GetWeakly_Coupled_Heat()) {
407  solvers[iMesh][HEAT_SOL]->SetFreeStream_Solution(config[iZone]);
408  solvers[iMesh][HEAT_SOL]->Postprocessing(geometry[iZone][iInst][iMesh], solvers[iMesh], config[iZone], iMesh);
409  }
410  }
411  }

414 void CDiscAdjFluidIteration::IterateDiscAdj(CGeometry**** geometry, CSolver***** solver, CConfig** config,
415  unsigned short iZone, unsigned short iInst, bool CrossTerm) {
416 
417  SU2_OMP_PARALLEL_(if(solver[iZone][iInst][MESH_0][ADJFLOW_SOL]->GetHasHybridParallel())) {
418 
419  /*--- Extract the adjoints of the conservative input variables and store them for the next iteration ---*/
420 
421  if (config[iZone]->GetFluidProblem()) {
422  solver[iZone][iInst][MESH_0][ADJFLOW_SOL]->ExtractAdjoint_Solution(geometry[iZone][iInst][MESH_0], config[iZone], CrossTerm);
423 
424  solver[iZone][iInst][MESH_0][ADJFLOW_SOL]->ExtractAdjoint_Variables(geometry[iZone][iInst][MESH_0], config[iZone]);
425  }
426  if (turbulent && !config[iZone]->GetFrozen_Visc_Disc()) {
427  solver[iZone][iInst][MESH_0][ADJTURB_SOL]->ExtractAdjoint_Solution(geometry[iZone][iInst][MESH_0], config[iZone], CrossTerm);
428  }
429  if (config[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
430  solver[iZone][iInst][MESH_0][ADJSPECIES_SOL]->ExtractAdjoint_Solution(geometry[iZone][iInst][MESH_0], config[iZone], CrossTerm);
431  }
432  if (config[iZone]->GetWeakly_Coupled_Heat()) {
433  solver[iZone][iInst][MESH_0][ADJHEAT_SOL]->ExtractAdjoint_Solution(geometry[iZone][iInst][MESH_0], config[iZone], CrossTerm);
434  }
435  if (config[iZone]->AddRadiation()) {
436  solver[iZone][iInst][MESH_0][ADJRAD_SOL]->ExtractAdjoint_Solution(geometry[iZone][iInst][MESH_0], config[iZone], CrossTerm);
437 
438  solver[iZone][iInst][MESH_0][ADJRAD_SOL]->ExtractAdjoint_Variables(geometry[iZone][iInst][MESH_0], config[iZone]);
439  }
440 
441  }
442  END_SU2_OMP_PARALLEL

447

480 void CDiscAdjFluidIteration::RegisterInput(CSolver***** solver, CGeometry**** geometry, CConfig** config,
481  unsigned short iZone, unsigned short iInst, RECORDING kind_recording) {
482 
483  SU2_OMP_PARALLEL_(if(solver[iZone][iInst][MESH_0][ADJFLOW_SOL]->GetHasHybridParallel())) {
484 
485  if (kind_recording == RECORDING::SOLUTION_VARIABLES || kind_recording == RECORDING::SOLUTION_AND_MESH) {
486  /*--- Register flow and turbulent variables as input ---*/
487 
488  if (config[iZone]->GetFluidProblem()) {
