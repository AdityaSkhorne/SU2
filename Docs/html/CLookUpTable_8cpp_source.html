142 void CLookUpTable::IdentifyUniqueEdges() {
143 
144  /* loop through elements and store the vector of element IDs (neighbors)
145  for each of the points in the table */
146 
147  vector<vector<unsigned long> > neighborElemsOfPoint;
148 
149  neighborElemsOfPoint.resize(n_points);
150  for (unsigned long iElem = 0; iElem < n_triangles; iElem++) {
151  /* loop over 3 points per triangle */
152  for (unsigned long iPoint = 0; iPoint < N_POINTS_TRIANGLE; iPoint++) {
153  /* get the global ID of the current point */
154  const unsigned long GlobalIndex = triangles[iElem][iPoint];
155 
156  /* add the current element ID to the neighbor list for this point */
157  neighborElemsOfPoint[GlobalIndex].push_back(iElem);
158  }
159  }
160 
161  /* remove duplicates from the neighboring element lists*/
162  vector<unsigned long>::iterator vecIt;
163  for (unsigned long iPoint = 0; iPoint < n_points; iPoint++) {
164  /* sort neighboring elements for each point */
165  sort(neighborElemsOfPoint[iPoint].begin(), neighborElemsOfPoint[iPoint].end());
166 
167  /* uniquify list of neighboring elements */
168  vecIt = unique(neighborElemsOfPoint[iPoint].begin(), neighborElemsOfPoint[iPoint].end());
169 
170  /* adjust size of vector */
171  neighborElemsOfPoint[iPoint].resize(vecIt - neighborElemsOfPoint[iPoint].begin());
172  }
173 
174  /* loop through all neighboring elements of each point and store
175  the point IDs that are neighboring points */
176  vector<vector<unsigned long> > neighborPointsOfPoint;
177  neighborPointsOfPoint.resize(n_points);
178  for (unsigned long iPoint = 0; iPoint < n_points; iPoint++) {
179  for (unsigned long iElem = 0; iElem < neighborElemsOfPoint[iPoint].size(); iElem++) {
180  /* loop over element points */
181  for (unsigned long jPoint = 0; jPoint < N_POINTS_TRIANGLE; jPoint++) {
182  /* get the global ID of the current point */
183  const unsigned long GlobalIndex = triangles[neighborElemsOfPoint[iPoint][iElem]][jPoint];
184 
185  /* add the current element ID to the neighbor list for this point */
186  if (GlobalIndex != iPoint) neighborPointsOfPoint[iPoint].push_back(GlobalIndex);
187  }
188  }
189  }
190 
191  /* remove duplicates from the neighboring points lists */
192  for (unsigned long iPoint = 0; iPoint < n_points; iPoint++) {
193  /* sort neighboring points for each point */
194  sort(neighborPointsOfPoint[iPoint].begin(), neighborPointsOfPoint[iPoint].end());
195 
196  /* uniquify list of neighboring elements */
197  vecIt = unique(neighborPointsOfPoint[iPoint].begin(), neighborPointsOfPoint[iPoint].end());
198 
199  /* adjust size of vector */
200  neighborPointsOfPoint[iPoint].resize(vecIt - neighborPointsOfPoint[iPoint].begin());
201  }
202 
203  /* loop through our point neighbors and fill the vector of the unique
204  point pairs making up each edge in the grid. We impose a requirement
205  that the smaller global index is in the first position and the larger
206  is in the second to make for a unique set, so there's no need to
207  remove duplicates. */
208  for (unsigned long iPoint = 0; iPoint < n_points; iPoint++) {
209  for (unsigned long jPoint = 0; jPoint < neighborPointsOfPoint[iPoint].size(); jPoint++) {
210  /* Store the neighbor index more clearly. */
211  const unsigned long GlobalIndex = neighborPointsOfPoint[iPoint][jPoint];
212 
213  /* Store the edge so that the lower index of the pair is always first. */
214  if (iPoint < GlobalIndex) {
215  vector<unsigned long> edge(2);
216  edge[0] = iPoint;
217  edge[1] = GlobalIndex;
218  edges.push_back(edge);
219  }
220  }
221  }
222 
223  /* Loop over our edges data structure. For the first point in each
224  pair, loop through the neighboring elements and store the two
225  elements that contain the second point in the edge ('left' and 'right' of the edge). */
226  edge_to_triangle.resize(edges.size());
227  for (unsigned long iEdge = 0; iEdge < edges.size(); iEdge++) {
228  /* Store the two points of the edge more clearly. */
229  const unsigned long iPoint = edges[iEdge][0];
230  const unsigned long jPoint = edges[iEdge][1];
231 
232  /* Loop over all neighboring elements to iPoint. */
233  for (unsigned long iElem = 0; iElem < neighborElemsOfPoint[iPoint].size(); iElem++) {
234  /* loop over 3 points per triangle */
235  for (unsigned long kPoint = 0; kPoint < N_POINTS_TRIANGLE; kPoint++) {
236  /* Get the global ID of the current point. */
237  const unsigned long GlobalIndex = triangles[neighborElemsOfPoint[iPoint][iElem]][kPoint];
238 
239  /* Add the current element ID to the neighbor list for this point. */
240  if (GlobalIndex == jPoint) edge_to_triangle[iEdge].push_back(neighborElemsOfPoint[iPoint][iElem]);
241  }
242  }
243  }
244 }

246 void CLookUpTable::ComputeInterpCoeffs(const string& name_prog, const string& name_enth) {
247  /* build KD tree for enthalpy, progress variable space */
248 
249  std::array<unsigned long, 3> next_triangle;
250 
251  const su2double* prog = GetDataP(name_prog);
252  const su2double* enth = GetDataP(name_enth);
253 
254  /* calculate weights for each triangle (basically a distance function) and
255  * build inverse interpolation matrices */
256  for (unsigned long i_triangle = 0; i_triangle < n_triangles; i_triangle++) {
257 
258  for (int p = 0; p < 3; p++) {
259  next_triangle[p] = triangles[i_triangle][p];
260  }
261 
262  su2activematrix prog_interp_mat_inv(3, 3);
263  GetInterpMatInv(prog, enth, next_triangle, prog_interp_mat_inv);
264  interp_mat_inv_prog_enth.push_back(prog_interp_mat_inv);
265  }
266 }

268 void CLookUpTable::GetInterpMatInv(const su2double* vec_x, const su2double* vec_y,
269  std::array<unsigned long,3>& point_ids, su2activematrix& interp_mat_inv) {
270  const unsigned int M = 3;
271  CSquareMatrixCM global_M(3);
272 
273  /* setup LHM matrix for the interpolation */
274  for (unsigned int i_point = 0; i_point < M; i_point++) {
275  su2double x = vec_x[point_ids[i_point]];
276  su2double y = vec_y[point_ids[i_point]];
277 
278  global_M(i_point,0) = SU2_TYPE::GetValue(1.0);
279  global_M(i_point,1) = SU2_TYPE::GetValue(x);
280  global_M(i_point,2) = SU2_TYPE::GetValue(y);
281  }
282 
283  global_M.Invert();
284  global_M.Transpose();
285 
286  for (unsigned int i=0; i<M; i++){
287  for (unsigned int j=0; j<M; j++){
288  interp_mat_inv[i][j] = global_M(i,j);
289  }
290  }
291 
292 }

294 unsigned long CLookUpTable::LookUp_ProgEnth(const string& val_name_var, su2double *val_var, su2double val_prog,
295  su2double val_enth, const string& name_prog, const string& name_enth) {
296  unsigned long exit_code = 0;
297 
298  if (val_name_var.compare("NULL") == 0) {
299  *val_var = 0.0;
300  exit_code = 0;
301  return exit_code;
302  }
303 
304  /* check if progress variable and enthalpy value is in table range */
305  if ((val_prog >= limits_table_prog[0] && val_prog <= limits_table_prog[1]) &&
306  (val_enth >= limits_table_enth[0] && val_enth <= limits_table_enth[1])){
307 
308  /* find the triangle that holds the (prog, enth) point */
309  unsigned long id_triangle = trap_map_prog_enth.GetTriangle(val_prog, val_enth);
310 
311  if (IsInTriangle(val_prog, val_enth, id_triangle, name_prog, name_enth)) {
312  /* get interpolation coefficients for point on triangle */
313  std::array<su2double,3> interp_coeffs{0};
314  GetInterpCoeffs(val_prog, val_enth, interp_mat_inv_prog_enth[id_triangle], interp_coeffs);
315 
316  /* first, copy the single triangle from the large triangle list*/
317  std::array<unsigned long,3> triangle{0};
318  for (int p = 0; p < 3; p++)
319  triangle[p] = triangles[id_triangle][p];
320 
321  *val_var = Interpolate(GetDataP(val_name_var), triangle, interp_coeffs);
322  exit_code = 0;
323  } else {
324  /* in bounding box but outside of table */
325  unsigned long nearest_neighbor = FindNearestNeighborOnHull(val_prog, val_enth, name_prog, name_enth);
326  *val_var = GetDataP(val_name_var)[nearest_neighbor];
327  exit_code = 1;
328  }
329  } else {
330  /* lookup is outside of table bounding box */
331  unsigned long nearest_neighbor = FindNearestNeighborOnHull(val_prog, val_enth, name_prog, name_enth);
332  *val_var = GetDataP(val_name_var)[nearest_neighbor];
333  exit_code = 1;
334  }
335  return exit_code;
336 }

352 unsigned long CLookUpTable::LookUp_ProgEnth(const vector<string>& val_names_var, vector<su2double*>& val_vars,
353  su2double val_prog, su2double val_enth, const string& name_prog,
354  const string& name_enth) {
355  unsigned long exit_code = 0;
356  unsigned long nearest_neighbor = 0;
357  unsigned long id_triangle;
358  std::array<su2double,3> interp_coeffs{0};
359  std::array<unsigned long,3> triangle{0};
360 
361  /* check if progress variable value is in progress variable table range
362  * and if enthalpy is in enthalpy table range */
363  if (val_prog >= limits_table_prog[0] && val_prog <= limits_table_prog[1] && val_enth >= limits_table_enth[0] &&
364  val_enth <= limits_table_enth[1]) {
365  /* if so, try to find the triangle that holds the (prog, enth) point */
366  id_triangle = trap_map_prog_enth.GetTriangle(val_prog, val_enth);
367 
368  /* check if point is inside a triangle (if table domain is non-rectangular,
369  * the previous range check might be true but the point could still be outside of the domain) */
370  if (IsInTriangle(val_prog, val_enth, id_triangle, name_prog, name_enth)) {
371  /* if so, get interpolation coefficients for point in the triangle */
372  GetInterpCoeffs(val_prog, val_enth, interp_mat_inv_prog_enth[id_triangle], interp_coeffs);
373 
374  /* exit_code 0 means point was in triangle */
375  exit_code = 0;
376 
377  } else {
378  /* if point is not inside a triangle (outside table domain) search nearest neighbor */
379  nearest_neighbor = FindNearestNeighborOnHull(val_prog, val_enth, name_prog, name_enth);
380  exit_code = 1;
381  }
382 
383  } else {
384  /* if point is outside of table ranges, find nearest neighbor */
385  nearest_neighbor = FindNearestNeighborOnHull(val_prog, val_enth, name_prog, name_enth);
386  exit_code = 1;
387  }
388 
389  /* loop over variable names and interpolate / get values */
390  for (long unsigned int i_var = 0; i_var < val_names_var.size(); ++i_var) {
391  if (val_names_var[i_var].compare("NULL") == 0) {
392  *val_vars[i_var] = 0.0;
393  } else {
394  if (exit_code == 0){
395 
396  /* first, copy the single triangle from the large triangle list*/
397  for (int p = 0; p < 3; p++)
398  triangle[p] = triangles[id_triangle][p];
399  *val_vars[i_var] = Interpolate(GetDataP(val_names_var[i_var]), triangle, interp_coeffs);
400  }
401  else
402  *val_vars[i_var] = GetDataP(val_names_var[i_var])[nearest_neighbor];
403  }
404  }
405  return exit_code;
406 }

408 void CLookUpTable::GetInterpCoeffs(su2double val_x, su2double val_y, su2activematrix& interp_mat_inv,
409  std::array<su2double,3>& interp_coeffs) {
410 
411  std::array<su2double,3> query_vector = {1,val_x,val_y};
412 
413  su2double d;
414  for (int i = 0; i < 3; i++) {
415  d = 0;
416  for (int j = 0; j < 3; j++) {
417  d = d + interp_mat_inv[i][j] * query_vector[j];
418  }
419  interp_coeffs[i] = d;
420  }
421 }

436 unsigned long CLookUpTable::FindNearestNeighborOnHull(su2double val_prog, su2double val_enth, const string& name_prog,
437  const string& name_enth) {
438  su2double min_distance = 1.e99;
439  su2double next_distance = 1.e99;
440  su2double next_prog_norm;
441  su2double next_enth_norm;
442  unsigned long neighbor_id = 0;
443 
444  const su2double* prog_table = GetDataP(name_prog);
445  const su2double* enth_table = GetDataP(name_enth);
446 
447  su2double norm_coeff_prog = 1. / (limits_table_prog[1] - limits_table_prog[0]);
448  su2double norm_coeff_enth = 1. / (limits_table_enth[1] - limits_table_enth[0]);
449  su2double val_prog_norm = val_prog / (limits_table_prog[1] - limits_table_prog[0]);
450  su2double val_enth_norm = val_enth / (limits_table_enth[1] - limits_table_enth[0]);
451 
452  for (unsigned long i_point = 0; i_point < n_hull_points; ++i_point) {
453  next_prog_norm = prog_table[hull[i_point]] * norm_coeff_prog;
454  next_enth_norm = enth_table[hull[i_point]] * norm_coeff_enth;
455 
456  next_distance = sqrt(pow(val_prog_norm - next_prog_norm, 2) + pow(val_enth_norm - next_enth_norm, 2));
457 
458  if (next_distance < min_distance) {
459  min_distance = next_distance;
460  neighbor_id = hull[i_point];
461  }
462  }
463  return neighbor_id;
464 }

466 bool CLookUpTable::IsInTriangle(su2double val_prog, su2double val_enth, unsigned long val_id_triangle, const string& name_prog,
467  const string& name_enth) {
468  su2double tri_prog_0 = GetDataP(name_prog)[triangles[val_id_triangle][0]];
469  su2double tri_enth_0 = GetDataP(name_enth)[triangles[val_id_triangle][0]];
470 
471  su2double tri_prog_1 = GetDataP(name_prog)[triangles[val_id_triangle][1]];
472  su2double tri_enth_1 = GetDataP(name_enth)[triangles[val_id_triangle][1]];
473 
474  su2double tri_prog_2 = GetDataP(name_prog)[triangles[val_id_triangle][2]];
475  su2double tri_enth_2 = GetDataP(name_enth)[triangles[val_id_triangle][2]];
476 
477  su2double area_tri = TriArea(tri_prog_0, tri_enth_0, tri_prog_1, tri_enth_1, tri_prog_2, tri_enth_2);
478 
479  su2double area_0 = TriArea(val_prog, val_enth, tri_prog_1, tri_enth_1, tri_prog_2, tri_enth_2);
480  su2double area_1 = TriArea(tri_prog_0, tri_enth_0, val_prog, val_enth, tri_prog_2, tri_enth_2);
481  su2double area_2 = TriArea(tri_prog_0, tri_enth_0, tri_prog_1, tri_enth_1, val_prog, val_enth);
482 
483  return (abs(area_tri - (area_0 + area_1 + area_2)) < area_tri * 1e-10);
484 }
