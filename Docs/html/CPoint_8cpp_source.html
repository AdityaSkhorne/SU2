57 void CPoint::FullAllocation(unsigned short imesh, const CConfig *config) {
58 
59  const auto npoint = GlobalIndex.size();
60 
61  /*--- Volumes ---*/
62 
63  Volume.resize(npoint) = su2double(0.0);
64  Periodic_Volume.resize(npoint) = su2double(0.0);
65 
66  if (config->GetTime_Marching() != TIME_MARCHING::STEADY) {
67  Volume_n.resize(npoint) = su2double(0.0);
68  Volume_nM1.resize(npoint) = su2double(0.0);
69  if (config->GetDynamic_Grid() && config->GetDiscrete_Adjoint()) {
70  Volume_Old.resize(npoint) = su2double(0.0);
71  Volume_n_Old.resize(npoint) = su2double(0.0);
72  Volume_nM1_Old.resize(npoint) = su2double(0.0);
73  }
74  }
75 
76  if (config->GetDiscrete_Adjoint()) {
77  AD_InputIndex.resize(npoint,nDim) = 0;
78  AD_OutputIndex.resize(npoint,nDim) = 0;
79  }
80 
81  /*--- Multigrid structures. ---*/
82  if (config->GetnMGLevels() > 0) {
83  Parent_CV.resize(npoint) = 0;
84  Agglomerate.resize(npoint) = false;
85  Agglomerate_Indirect.resize(npoint) = false;
86  /*--- The finest grid does not have children CV's. ---*/
87  if (imesh != MESH_0) {
88  nChildren_CV.resize(npoint) = 0;
89  Children_CV.resize(npoint);
90  }
91  }
92 
93  /*--- Identify boundaries, physical boundaries (not send-receive condition), detect if
94  * an element belong to the domain or it must be computed with other processor. ---*/
95  Domain.resize(npoint) = true;
96  Boundary.resize(npoint) = false;
97  SolidBoundary.resize(npoint) = false;
98  ViscousBoundary.resize(npoint) = false;
99  PhysicalBoundary.resize(npoint) = false;
100  PeriodicBoundary.resize(npoint) = false;
101 
102  Vertex.resize(npoint);
103 
104  /*--- For smoothing the numerical grid coordinates ---*/
105  if (config->GetSmoothNumGrid()) {
106  Coord_Old.resize(npoint,nDim) = su2double(0.0);
107  Coord_Sum.resize(npoint,nDim) = su2double(0.0);
108  }
109 
110  /*--- Storage of grid velocities for dynamic meshes. ---*/
111 
112  if (config->GetDynamic_Grid()) {
113  GridVel.resize(npoint,nDim) = su2double(0.0);
114 
115  /*--- Grid velocity gradients are needed for the continuous adjoint. ---*/
116  if (config->GetContinuous_Adjoint())
117  GridVel_Grad.resize(npoint,nDim,nDim,0.0);
118 
119  /*--- Structures for storing old node coordinates for computing grid
120  * velocities via finite differencing with dynamically deforming meshes. ---*/
121  /*--- In the case of CMeshSolver, these coordinates are stored as solutions to the mesh problem. ---*/
122  if (config->GetGrid_Movement() && (config->GetTime_Marching() != TIME_MARCHING::STEADY)) {
123  Coord_n.resize(npoint,nDim) = su2double(0.0);
124  Coord_p1.resize(npoint,nDim) = su2double(0.0);
125  Coord_n1.resize(npoint,nDim) = su2double(0.0);
126  if (Coord_Old.empty()) Coord_Old.resize(npoint,nDim) = su2double(0.0);
127  }
128  }
129 
130  if (config->GetTopology_Optimization())
131  Aux_Var.resize(npoint) = su2double(0.5);
132 
133  /*--- Other geometric properties of the CV's required by numerical methods. ---*/
134  nNeighbor.resize(npoint) = 0;
135  MaxLength.resize(npoint) = su2double(0.0);
136  Curvature.resize(npoint) = su2double(0.0);
137 
138  Wall_Distance.resize(npoint) = su2double(0.0);
139  ClosestWall_Rank.resize(npoint) = -1;
140  ClosestWall_Zone.resize(npoint) = numeric_limits<unsigned short>::max();
141  ClosestWall_Marker = ClosestWall_Zone;
142  ClosestWall_Elem.resize(npoint) = numeric_limits<unsigned long>::max();
143 
144  RoughnessHeight.resize(npoint) = su2double(0.0);
145  SharpEdge_Distance.resize(npoint) = su2double(0.0);
146 
147 }
