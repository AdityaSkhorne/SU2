40 void CCentSca_Heat::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i,
41  su2double **val_Jacobian_j, CConfig *config) {
42  AD::StartPreacc();
43  AD::SetPreaccIn(V_i, nDim+3); AD::SetPreaccIn(V_j, nDim+3);
44  AD::SetPreaccIn(Temp_i); AD::SetPreaccIn(Temp_j);
45  AD::SetPreaccIn(Und_Lapl_i, nVar); AD::SetPreaccIn(Und_Lapl_j, nVar);
46  AD::SetPreaccIn(Normal, nDim);
47  if (dynamic_grid) {
48  AD::SetPreaccIn(GridVel_i, nDim); AD::SetPreaccIn(GridVel_j, nDim);
49  }
50 
51  /*--- Primitive variables at point i and j ---*/
52 
53  Pressure_i = V_i[0]; Pressure_j = V_j[0];
54  DensityInc_i = V_i[nDim+2]; DensityInc_j = V_j[nDim+2];
55  BetaInc2_i = V_i[nDim+3]; BetaInc2_j = V_j[nDim+3];
56 
57  /*--- Projected velocities at the current edge ---*/
58 
59  su2double ProjVelocity_i = 0.0;
60  su2double ProjVelocity_j = 0.0;
61 
62  if (dynamic_grid) {
63  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
64  su2double Velocity_i = V_i[iDim+1] - GridVel_i[iDim];
65  su2double Velocity_j = V_j[iDim+1] - GridVel_j[iDim];
66  ProjVelocity_i += Velocity_i*Normal[iDim];
67  ProjVelocity_j += Velocity_j*Normal[iDim];
68  }
69  }
70  else {
71  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
72  ProjVelocity_i += V_i[iDim+1]*Normal[iDim];
73  ProjVelocity_j += V_j[iDim+1]*Normal[iDim];
74  }
75  }
76 
77  /*--- Computing the second order centered scheme part ---*/
78 
79  const su2double ProjVelocity = 0.5*(ProjVelocity_i+ProjVelocity_j);
80 
81  val_residual[0] = 0.5*(Temp_i + Temp_j)*ProjVelocity;
82 
83  if (implicit) {
84  val_Jacobian_i[0][0] = 0.5*ProjVelocity;
85  val_Jacobian_j[0][0] = 0.5*ProjVelocity;
86  }
87 
88  /*--- Adding artificial dissipation to stabilize the centered scheme ---*/
89 
90  const su2double Diff_Lapl = Und_Lapl_i[0]-Und_Lapl_j[0];
91  const su2double Area2 = GeometryToolbox::SquaredNorm(nDim, Normal);
92 
93  const su2double SoundSpeed_i = sqrt(pow(ProjVelocity_i,2) + (BetaInc2_i/DensityInc_i)*Area2);
94  const su2double SoundSpeed_j = sqrt(pow(ProjVelocity_j,2) + (BetaInc2_j/DensityInc_j)*Area2);
95 
96  const su2double Local_Lambda_i = fabs(ProjVelocity_i)+SoundSpeed_i;
97  const su2double Local_Lambda_j = fabs(ProjVelocity_j)+SoundSpeed_j;
98  const su2double MeanLambda = 0.5*(Local_Lambda_i+Local_Lambda_j);
99 
100  val_residual[0] += -Param_Kappa_4*Diff_Lapl*MeanLambda;
101 
102  if (implicit) {
103  const su2double cte_0 = Param_Kappa_4*su2double(Neighbor_i+1)*MeanLambda;
104  const su2double cte_1 = Param_Kappa_4*su2double(Neighbor_j+1)*MeanLambda;
105 
106  val_Jacobian_i[0][0] += cte_0;
107  val_Jacobian_j[0][0] -= cte_1;
108  }
109 
110  AD::SetPreaccOut(val_residual[0]);
111  AD::EndPreacc();
112 
113 }

123 void CUpwSca_Heat::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i,
124  su2double **val_Jacobian_j, CConfig *config) {
125 
126  AD::StartPreacc();
127  AD::SetPreaccIn(V_i, nDim+1); AD::SetPreaccIn(V_j, nDim+1);
128  AD::SetPreaccIn(Temp_i); AD::SetPreaccIn(Temp_j);
129  AD::SetPreaccIn(Normal, nDim);
130  if (dynamic_grid) {
131  AD::SetPreaccIn(GridVel_i, nDim); AD::SetPreaccIn(GridVel_j, nDim);
132  }
133 
134  su2double q_ij = 0.0;
135 
136  if (dynamic_grid) {
137  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
138  const su2double Velocity_i = V_i[iDim+1] - GridVel_i[iDim];
139  const su2double Velocity_j = V_j[iDim+1] - GridVel_j[iDim];
140  q_ij += 0.5*(Velocity_i+Velocity_j)*Normal[iDim];
141  }
142  }
143  else {
144  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
145  q_ij += 0.5*(V_i[iDim+1]+V_j[iDim+1])*Normal[iDim];
146  }
147  }
148 
149  const su2double a0 = 0.5*(q_ij+fabs(q_ij));
150  const su2double a1 = 0.5*(q_ij-fabs(q_ij));
151 
152  val_residual[0] = a0*Temp_i+a1*Temp_j;
153 
154  if (implicit) {
155  val_Jacobian_i[0][0] = a0;
156  val_Jacobian_j[0][0] = a1;
157  }
158 
159  AD::SetPreaccOut(val_residual[0]);
160  AD::EndPreacc();
161 
162 }
