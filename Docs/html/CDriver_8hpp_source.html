742  /*!
743  * \brief Set the solution of all solvers (adjoint or primal) in a zone.
744  * \param[in] iZone - Index of the zone.
745  * \param[in] adjoint - True to consider adjoint solvers instead of primal.
746  * \param[in] solution - Solution object with interface (iPoint,iVar).
747  * \tparam Old - If true set "old solutions" instead.
748  */
749  template<class Container, bool Old = false>
750  void SetAllSolutions(unsigned short iZone, bool adjoint, const Container& solution) {
751  const auto nPoint = geometry_container[iZone][INST_0][MESH_0]->GetnPoint();
752  for (auto iSol = 0u, offset = 0u; iSol < MAX_SOLS; ++iSol) {
753  auto solver = solver_container[iZone][INST_0][MESH_0][iSol];
754  if (!(solver && (solver->GetAdjoint() == adjoint))) continue;
755  for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint)
756  for (auto iVar = 0ul; iVar < solver->GetnVar(); ++iVar)
757  if (!Old) solver->GetNodes()->SetSolution(iPoint, iVar, solution(iPoint,offset+iVar));
758  else solver->GetNodes()->SetSolution_Old(iPoint, iVar, solution(iPoint,offset+iVar));
759  offset += solver->GetnVar();
760  }
761  }

768  SetAllSolutions<Container,true>(iZone, adjoint, solution);

775  * \param[out] solution - Solution object with interface (iPoint,iVar).
776  */
777  template<class Container>
778  void GetAllSolutions(unsigned short iZone, bool adjoint, Container& solution) const {
779  const auto nPoint = geometry_container[iZone][INST_0][MESH_0]->GetnPoint();
780  for (auto iSol = 0u, offset = 0u; iSol < MAX_SOLS; ++iSol) {
781  auto solver = solver_container[iZone][INST_0][MESH_0][iSol];

784  for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint)
