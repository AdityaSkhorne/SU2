312 vector<passivedouble> CDriver::GetVertexHeatFluxes(unsigned short iMarker, unsigned long iVertex) const {
313 
314  unsigned long iPoint;
315  unsigned short iDim;
316  su2double Prandtl_Lam = config_container[ZONE_0]->GetPrandtl_Lam();
317  su2double Gas_Constant = config_container[ZONE_0]->GetGas_ConstantND();
318  su2double Gamma = config_container[ZONE_0]->GetGamma();
319  su2double Gamma_Minus_One = Gamma - 1.0;
320  su2double Cp = (Gamma / Gamma_Minus_One) * Gas_Constant;
321  su2double laminar_viscosity, thermal_conductivity;
322  vector<su2double> GradT (3,0.0);
323  vector<su2double> HeatFlux (3,0.0);
324  vector<passivedouble> HeatFluxPassive (3,0.0);
325 
326  bool compressible = (config_container[ZONE_0]->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE);
327 
328  iPoint = geometry_container[ZONE_0][INST_0][MESH_0]->vertex[iMarker][iVertex]->GetNode();
329 
330  if(compressible){
331  laminar_viscosity = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
332  thermal_conductivity = Cp * (laminar_viscosity/Prandtl_Lam);
333  for(iDim=0; iDim < nDim; iDim++){
334  GradT[iDim] = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->GetNodes()->GetGradient_Primitive(iPoint, 0, iDim);
335  HeatFlux[iDim] = -thermal_conductivity*GradT[iDim];
336  }
337  }
338 
339  HeatFluxPassive[0] = SU2_TYPE::GetValue(HeatFlux[0]);
340  HeatFluxPassive[1] = SU2_TYPE::GetValue(HeatFlux[1]);
341  HeatFluxPassive[2] = SU2_TYPE::GetValue(HeatFlux[2]);
342 
343  return HeatFluxPassive;
344 }

346 passivedouble CDriver::GetVertexNormalHeatFlux(unsigned short iMarker, unsigned long iVertex) const{
347 
348  unsigned long iPoint;
349  unsigned short iDim;
350  su2double vertexWallHeatFlux;
351  su2double Prandtl_Lam = config_container[ZONE_0]->GetPrandtl_Lam();
352  su2double Gas_Constant = config_container[ZONE_0]->GetGas_ConstantND();
353  su2double Gamma = config_container[ZONE_0]->GetGamma();
354  su2double Gamma_Minus_One = Gamma - 1.0;
355  su2double Cp = (Gamma / Gamma_Minus_One) * Gas_Constant;
356  su2double Area;
357  su2double laminar_viscosity, thermal_conductivity, dTdn;
358  su2double *Normal, GradT[3] = {0.0,0.0,0.0}, UnitNormal[3] = {0.0,0.0,0.0};
359 
360  bool compressible = (config_container[ZONE_0]->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE);
361 
362  vertexWallHeatFlux = 0.0;
363  dTdn = 0.0;
364 
365  iPoint = geometry_container[ZONE_0][INST_0][MESH_0]->vertex[iMarker][iVertex]->GetNode();
366 
367  if(geometry_container[ZONE_0][INST_0][MESH_0]->nodes->GetDomain(iPoint) && compressible){
368  Normal = geometry_container[ZONE_0][INST_0][MESH_0]->vertex[iMarker][iVertex]->GetNormal();
369 
370  Area = GeometryToolbox::Norm(nDim, Normal);
371 
372  for (iDim = 0; iDim < nDim; iDim++)
373  UnitNormal[iDim] = Normal[iDim]/Area;
374 
375  laminar_viscosity = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
376  thermal_conductivity = Cp * (laminar_viscosity/Prandtl_Lam);
377  /*Compute wall heat flux (normal to the wall) based on computed temperature gradient*/
378  for(iDim=0; iDim < nDim; iDim++){
379  GradT[iDim] = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->GetNodes()->GetGradient_Primitive(iPoint, 0, iDim);
380  dTdn += GradT[iDim]*UnitNormal[iDim];
381  }
382 
383  vertexWallHeatFlux = -thermal_conductivity*dTdn;
384  }
385 
386  return SU2_TYPE::GetValue(vertexWallHeatFlux);
387 }
