83 CNumerics::ResidualType<> CUpwAUSM_NEMO::ComputeResidual(const CConfig *config) {
84 
85  /*--- Compute geometric quantities ---*/
86  Area = GeometryToolbox::Norm(nDim, Normal);
87 
88  for (unsigned short iDim = 0; iDim < nDim; iDim++)
89  UnitNormal[iDim] = Normal[iDim]/Area;
90 
91  /*--- Pull stored primitive variables ---*/
92  // Primitives: [rho1,...,rhoNs, T, Tve, u, v, w, P, rho, h, a, c]
93  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
94  rhos_i[iSpecies] = V_i[RHOS_INDEX+iSpecies];
95  rhos_j[iSpecies] = V_j[RHOS_INDEX+iSpecies];
96  }
97  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
98  u_i[iDim] = V_i[VEL_INDEX+iDim];
99  u_j[iDim] = V_j[VEL_INDEX+iDim];
100  }
101 
102  P_i = V_i[P_INDEX]; P_j = V_j[P_INDEX];
103  h_i = V_i[H_INDEX]; h_j = V_j[H_INDEX];
104  a_i = V_i[A_INDEX]; a_j = V_j[A_INDEX];
105  rho_i = V_i[RHO_INDEX]; rho_j = V_j[RHO_INDEX];
106 
107  e_ve_i = 0; e_ve_j = 0;
108  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
109  e_ve_i += (V_i[RHOS_INDEX+iSpecies]*eve_i[iSpecies])/rho_i;
110  e_ve_j += (V_j[RHOS_INDEX+iSpecies]*eve_j[iSpecies])/rho_j;
111  }
112 
113  /*--- Projected velocities ---*/
114  ProjVel_i = GeometryToolbox::DotProduct(nDim, u_i, UnitNormal);
115  ProjVel_j = GeometryToolbox::DotProduct(nDim, u_j, UnitNormal);
116 
117  /*--- Calculate L/R Mach numbers ---*/
118  su2double mL = ProjVel_i/a_i;
119  su2double mR = ProjVel_j/a_j;
120 
121  /*--- Calculate split numerical fluxes ---*/
122  su2double mLP, mRM, pLP, pRM;
123 
124  if (fabs(mL) <= 1.0) mLP = 0.25*(mL+1.0)*(mL+1.0);
125  else mLP = 0.5*(mL+fabs(mL));
126 
127  if (fabs(mR) <= 1.0) mRM = -0.25*(mR-1.0)*(mR-1.0);
128  else mRM = 0.5*(mR-fabs(mR));
129 
130  su2double mF = mLP + mRM;
131 
132  if (fabs(mL) <= 1.0) pLP = 0.25*P_i*(mL+1.0)*(mL+1.0)*(2.0-mL);
133  else pLP = 0.5*P_i*(mL+fabs(mL))/mL;
134 
135  if (fabs(mR) <= 1.0) pRM = 0.25*P_j*(mR-1.0)*(mR-1.0)*(2.0+mR);
136  else pRM = 0.5*P_j*(mR-fabs(mR))/mR;
137 
138  su2double pF = pLP + pRM;
139  su2double Phi = fabs(mF);
140 
141  /*--- Assign left & right convective vectors ---*/
142  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
143  FcL[iSpecies] = rhos_i[iSpecies]*a_i;
144  FcR[iSpecies] = rhos_j[iSpecies]*a_j;
145  }
146  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
147  FcL[nSpecies+iDim] = rho_i*a_i*u_i[iDim];
148  FcR[nSpecies+iDim] = rho_j*a_j*u_j[iDim];
149  }
150  FcL[nSpecies+nDim] = rho_i*a_i*h_i;
151  FcR[nSpecies+nDim] = rho_j*a_j*h_j;
152  FcL[nSpecies+nDim+1] = rho_i*a_i*e_ve_i;
153  FcR[nSpecies+nDim+1] = rho_j*a_j*e_ve_j;
154 
155  /*--- Compute numerical flux ---*/
156  for (unsigned short iVar = 0; iVar < nVar; iVar++)
157  Flux[iVar] = 0.5*((mF+Phi)*FcL[iVar]+(mF-Phi)*FcR[iVar])*Area;
158 
159  for (unsigned short iDim = 0; iDim < nDim; iDim++)
160  Flux[nSpecies+iDim] += pF*UnitNormal[iDim]*Area;
161 
162  /*--- Populate the AUSM Jacobian, if necessary. ---*/
163  if (implicit){
164 
165  const auto& Ms = fluidmodel->GetSpeciesMolarMass();
166  const auto& Cvtr = fluidmodel->GetSpeciesCvTraRot();
167  const su2double Ru = 1000.0*UNIVERSAL_GAS_CONSTANT;
168  rhoCvtr_i = V_i[RHOCVTR_INDEX];
169  rhoCvtr_j = V_j[RHOCVTR_INDEX];
170 
171  /*--- Initialize the Jacobians ---*/
172  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
173  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
174  Jacobian_i[iVar][jVar] = 0.0;
175  Jacobian_j[iVar][jVar] = 0.0;
176  }
177  }
178 
179  if (mF >= 0.0) FcLR = FcL;
180  else FcLR = FcR;
181 
182  /*--- Sound speed derivatives: Species density ---*/
183  for (unsigned short iSpecies = 0; iSpecies < nEl; iSpecies++) {
184  daL[iSpecies] = 1.0/(2.0*a_i*rho_i) * (1+dPdU_i[nSpecies+nDim])*(dPdU_i[iSpecies] - P_i/rho_i);
185  daR[iSpecies] = 1.0/(2.0*a_j*rho_j) * (1+dPdU_j[nSpecies+nDim])*(dPdU_j[iSpecies] - P_j/rho_j);
186  }
187 
188  for (unsigned short iSpecies = nEl; iSpecies < nHeavy; iSpecies++) {
189  daL[iSpecies] = 1.0/(2.0*a_i) * (1/rhoCvtr_i*(Ru/Ms[iSpecies] - Cvtr[iSpecies]*dPdU_i[nSpecies+nDim])*P_i/rho_i
190  + 1.0/rho_i*(1.0+dPdU_i[nSpecies+nDim])*(dPdU_i[iSpecies] - P_i/rho_i));
191  daR[iSpecies] = 1.0/(2.0*a_j) * (1/rhoCvtr_j*(Ru/Ms[iSpecies] - Cvtr[iSpecies]*dPdU_j[nSpecies+nDim])*P_j/rho_j
192  + 1.0/rho_j*(1.0+dPdU_j[nSpecies+nDim])*(dPdU_j[iSpecies] - P_j/rho_j));
193  }
194 
195  /*--- Sound speed derivatives: Momentum ---*/
196  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
197  daL[nSpecies+iDim] = -1.0/(2.0*rho_i*a_i) * ((1.0+dPdU_i[nSpecies+nDim])*dPdU_i[nSpecies+nDim])*u_i[iDim];
198  daR[nSpecies+iDim] = -1.0/(2.0*rho_j*a_j) * ((1.0+dPdU_j[nSpecies+nDim])*dPdU_j[nSpecies+nDim])*u_j[iDim];
199  }
200 
201  /*--- Sound speed derivatives: Energy ---*/
202  daL[nSpecies+nDim] = 1.0/(2.0*rho_i*a_i) * ((1.0+dPdU_i[nSpecies+nDim])*dPdU_i[nSpecies+nDim]);
203  daR[nSpecies+nDim] = 1.0/(2.0*rho_j*a_j) * ((1.0+dPdU_j[nSpecies+nDim])*dPdU_j[nSpecies+nDim]);
204 
205  /*--- Sound speed derivatives: Vib-el energy ---*/
206  daL[nSpecies+nDim+1] = 1.0/(2.0*rho_i*a_i) * ((1.0+dPdU_i[nSpecies+nDim])*dPdU_i[nSpecies+nDim+1]);
207  daR[nSpecies+nDim+1] = 1.0/(2.0*rho_j*a_j) * ((1.0+dPdU_j[nSpecies+nDim])*dPdU_j[nSpecies+nDim+1]);
208 
209  /*--- Left state Jacobian ---*/
210  if (mF >= 0) {
211 
212  /*--- Jacobian contribution: dFc terms ---*/
213  for (auto iVar = 0u; iVar < nSpecies+nDim; iVar++) {
214  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
215  Jacobian_i[iVar][jVar] += mF * FcL[iVar]/a_i * daL[jVar];
216  }
217  Jacobian_i[iVar][iVar] += mF * a_i;
218  }
219  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
220  Jacobian_i[nSpecies+nDim][iSpecies] += mF * (dPdU_i[iSpecies]*a_i + rho_i*h_i*daL[iSpecies]);
221  }
222  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
223  Jacobian_i[nSpecies+nDim][nSpecies+iDim] += mF * (-dPdU_i[nSpecies+nDim]*u_i[iDim]*a_i + rho_i*h_i*daL[nSpecies+iDim]);
224  }
225  Jacobian_i[nSpecies+nDim][nSpecies+nDim] += mF * ((1.0+dPdU_i[nSpecies+nDim])*a_i + rho_i*h_i*daL[nSpecies+nDim]);
226  Jacobian_i[nSpecies+nDim][nSpecies+nDim+1] += mF * (dPdU_i[nSpecies+nDim+1]*a_i + rho_i*h_i*daL[nSpecies+nDim+1]);
227  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
228  Jacobian_i[nSpecies+nDim+1][jVar] += mF * FcL[nSpecies+nDim+1]/a_i * daL[jVar];
229  }
230  Jacobian_i[nSpecies+nDim+1][nSpecies+nDim+1] += mF * a_i;
231  }
232 
233  /*--- Calculate derivatives of the split pressure flux ---*/
234  if ( (mF >= 0) || ((mF < 0)&&(fabs(mF) <= 1.0)) ) {
235  if (fabs(mL) <= 1.0) {
236 
237  /*--- Mach number ---*/
238  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
239  dmLP[iSpecies] = 0.5*(mL+1.0) * (-ProjVel_i/(rho_i*a_i) - ProjVel_i*daL[iSpecies]/(a_i*a_i));
240  for (unsigned short iDim = 0; iDim < nDim; iDim++)
241  dmLP[nSpecies+iDim] = 0.5*(mL+1.0) * (-ProjVel_i/(a_i*a_i) * daL[nSpecies+iDim] + UnitNormal[iDim]/(rho_i*a_i));
242  dmLP[nSpecies+nDim] = 0.5*(mL+1.0) * (-ProjVel_i/(a_i*a_i) * daL[nSpecies+nDim]);
243  dmLP[nSpecies+nDim+1] = 0.5*(mL+1.0) * (-ProjVel_i/(a_i*a_i) * daL[nSpecies+nDim+1]);
244 
245  /*--- Pressure ---*/
246  for(unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
247  dpLP[iSpecies] = 0.25*(mL+1.0) * (dPdU_i[iSpecies]*(mL+1.0)*(2.0-mL)
248  + P_i*(-ProjVel_i/(rho_i*a_i)
249  -ProjVel_i*daL[iSpecies]/(a_i*a_i))*(3.0-3.0*mL));
250  for (unsigned short iDim = 0; iDim < nDim; iDim++)
251  dpLP[nSpecies+iDim] = 0.25*(mL+1.0) * (-u_i[iDim]*dPdU_i[nSpecies+nDim]*(mL+1.0)*(2.0-mL)
252  + P_i*( -ProjVel_i/(a_i*a_i) * daL[nSpecies+iDim]
253  + UnitNormal[iDim]/(rho_i*a_i))*(3.0-3.0*mL));
254  dpLP[nSpecies+nDim] = 0.25*(mL+1.0) * (dPdU_i[nSpecies+nDim]*(mL+1.0)*(2.0-mL)
255  + P_i*(-ProjVel_i/(a_i*a_i) * daL[nSpecies+nDim])*(3.0-3.0*mL));
256  dpLP[nSpecies+nDim+1] = 0.25*(mL+1.0) * (dPdU_i[nSpecies+nDim+1]*(mL+1.0)*(2.0-mL)
257  + P_i*(-ProjVel_i/(a_i*a_i) * daL[nSpecies+nDim+1])*(3.0-3.0*mL));
258  } else {
259 
260  /*--- Mach number ---*/
261  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
262  dmLP[iSpecies] = -ProjVel_i/(rho_i*a_i) - ProjVel_i*daL[iSpecies]/(a_i*a_i);
263  for (unsigned short iDim = 0; iDim < nDim; iDim++)
264  dmLP[nSpecies+iDim] = -ProjVel_i/(a_i*a_i) * daL[nSpecies+iDim] + UnitNormal[iDim]/(rho_i*a_i);
265  dmLP[nSpecies+nDim] = -ProjVel_i/(a_i*a_i) * daL[nSpecies+nDim];
266  dmLP[nSpecies+nDim+1] = -ProjVel_i/(a_i*a_i) * daL[nSpecies+nDim+1];
267 
268  /*--- Pressure ---*/
269  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
270  dpLP[iSpecies] = dPdU_i[iSpecies];
271  for (unsigned short iDim = 0; iDim < nDim; iDim++)
272  dpLP[nSpecies+iDim] = (-u_i[iDim]*dPdU_i[nSpecies+nDim]);
273  dpLP[nSpecies+nDim] = dPdU_i[nSpecies+nDim];
274  dpLP[nSpecies+nDim+1] = dPdU_i[nSpecies+nDim+1];
275  }
276 
277  /*--- dM contribution ---*/
278  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
279  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
280  Jacobian_i[iVar][jVar] += dmLP[jVar]*FcLR[iVar];
281  }
282  }
283 
284  /*--- Jacobian contribution: dP terms ---*/
285  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
286  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
287  Jacobian_i[nSpecies+iDim][iVar] += dpLP[iVar]*UnitNormal[iDim];
288  }
289  }
290  }
291 
292  /*--- Right state Jacobian ---*/
293  if (mF < 0) {
294 
295  /*--- Jacobian contribution: dFc terms ---*/
296  for (auto iVar = 0u; iVar < nSpecies+nDim; iVar++) {
297  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
298  Jacobian_j[iVar][jVar] += mF * FcR[iVar]/a_j * daR[jVar];
299  }
300  Jacobian_j[iVar][iVar] += mF * a_j;
301  }
302  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
303  Jacobian_j[nSpecies+nDim][iSpecies] += mF * (dPdU_j[iSpecies]*a_j + rho_j*h_j*daR[iSpecies]);
304  }
305  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
306  Jacobian_j[nSpecies+nDim][nSpecies+iDim] += mF * (-dPdU_j[nSpecies+nDim]*u_j[iDim]*a_j + rho_j*h_j*daR[nSpecies+iDim]);
307  }
308  Jacobian_j[nSpecies+nDim][nSpecies+nDim] += mF * ((1.0+dPdU_j[nSpecies+nDim])*a_j + rho_j*h_j*daR[nSpecies+nDim]);
309  Jacobian_j[nSpecies+nDim][nSpecies+nDim+1] += mF * (dPdU_j[nSpecies+nDim+1]*a_j + rho_j*h_j*daR[nSpecies+nDim+1]);
310  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
311  Jacobian_j[nSpecies+nDim+1][jVar] += mF * FcR[nSpecies+nDim+1]/a_j * daR[jVar];
312  }
313  Jacobian_j[nSpecies+nDim+1][nSpecies+nDim+1] += mF * a_j;
314  }
315 
316  /*--- Calculate derivatives of the split pressure flux ---*/
317  if ( (mF < 0) || ((mF >= 0)&&(fabs(mF) <= 1.0)) ) {
318  if (fabs(mR) <= 1.0) {
319 
320  /*--- Mach ---*/
321  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
322  dmRM[iSpecies] = -0.5*(mR-1.0) * (-ProjVel_j/(rho_j*a_j) - ProjVel_j*daR[iSpecies]/(a_j*a_j));
323  for (unsigned short iDim = 0; iDim < nDim; iDim++)
324  dmRM[nSpecies+iDim] = -0.5*(mR-1.0) * (-ProjVel_j/(a_j*a_j) * daR[nSpecies+iDim] + UnitNormal[iDim]/(rho_j*a_j));
325  dmRM[nSpecies+nDim] = -0.5*(mR-1.0) * (-ProjVel_j/(a_j*a_j) * daR[nSpecies+nDim]);
326  dmRM[nSpecies+nDim+1] = -0.5*(mR-1.0) * (-ProjVel_j/(a_j*a_j) * daR[nSpecies+nDim+1]);
327 
328  /*--- Pressure ---*/
329  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
330  dpRM[iSpecies] = 0.25*(mR-1.0) * (dPdU_j[iSpecies]*(mR-1.0)*(2.0+mR)
331  + P_j*(-ProjVel_j/(rho_j*a_j)
332  -ProjVel_j*daR[iSpecies]/(a_j*a_j))*(3.0+3.0*mR));
333  for (unsigned short iDim = 0; iDim < nDim; iDim++)
334  dpRM[nSpecies+iDim] = 0.25*(mR-1.0) * ((-u_j[iDim]*dPdU_j[nSpecies+nDim])*(mR-1.0)*(2.0+mR)
335  + P_j*( -ProjVel_j/(a_j*a_j) * daR[nSpecies+iDim]
336  + UnitNormal[iDim]/(rho_j*a_j))*(3.0+3.0*mR));
337  dpRM[nSpecies+nDim] = 0.25*(mR-1.0) * (dPdU_j[nSpecies+nDim]*(mR-1.0)*(2.0+mR)
338  + P_j*(-ProjVel_j/(a_j*a_j)*daR[nSpecies+nDim])*(3.0+3.0*mR));
339  dpRM[nSpecies+nDim+1] = 0.25*(mR-1.0) * (dPdU_j[nSpecies+nDim+1]*(mR-1.0)*(2.0+mR)
340  + P_j*(-ProjVel_j/(a_j*a_j) * daR[nSpecies+nDim+1])*(3.0+3.0*mR));
341 
342  } else {
343 
344  /*--- Mach ---*/
345  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
346  dmRM[iSpecies] = -ProjVel_j/(rho_j*a_j) - ProjVel_j*daR[iSpecies]/(a_j*a_j);
347  for (unsigned short iDim = 0; iDim < nDim; iDim++)
348  dmRM[nSpecies+iDim] = -ProjVel_j/(a_j*a_j) * daR[nSpecies+iDim] + UnitNormal[iDim]/(rho_j*a_j);
349  dmRM[nSpecies+nDim] = -ProjVel_j/(a_j*a_j) * daR[nSpecies+nDim];
350  dmRM[nSpecies+nDim+1] = -ProjVel_j/(a_j*a_j) * daR[nSpecies+nDim+1];
351 
352  /*--- Pressure ---*/
353  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
354  dpRM[iSpecies] = dPdU_j[iSpecies];
355  for (unsigned short iDim = 0; iDim < nDim; iDim++)
356  dpRM[nSpecies+iDim] = -u_j[iDim]*dPdU_j[nSpecies+nDim];
357  dpRM[nSpecies+nDim] = dPdU_j[nSpecies+nDim];
358  dpRM[nSpecies+nDim+1] = dPdU_j[nSpecies+nDim+1];
359  }
360 
361  /*--- Jacobian contribution: dM terms ---*/
362  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
363  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
364  Jacobian_j[iVar][jVar] += dmRM[jVar] * FcLR[iVar];
365  }
366  }
367 
368  /*--- Jacobian contribution: dP terms ---*/
369  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
370  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
371  Jacobian_j[nSpecies+iDim][iVar] += dpRM[iVar]*UnitNormal[iDim];
372  }
