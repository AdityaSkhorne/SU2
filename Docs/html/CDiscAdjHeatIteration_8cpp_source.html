31 void CDiscAdjHeatIteration::Preprocess(COutput* output, CIntegration**** integration, CGeometry**** geometry,
32  CSolver***** solver, CNumerics****** numerics, CConfig** config,
33  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
34  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
35 
36  const auto TimeIter = config[val_iZone]->GetTimeIter();
37  const bool dual_time_1st = (config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST);
38  const bool dual_time_2nd = (config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
39  const bool dual_time = (dual_time_1st || dual_time_2nd);
40 
41  auto solvers = solver[val_iZone][val_iInst];
42 
43  /*--- For the unsteady adjoint, load direct solutions from restart files. ---*/
44 
45  if (config[val_iZone]->GetTime_Marching() != TIME_MARCHING::STEADY) {
46  const int Direct_Iter = static_cast<int>(config[val_iZone]->GetUnst_AdjointIter()) - static_cast<int>(TimeIter) - 2 + dual_time;
47 
48  /*--- For dual-time stepping we want to load the already converged solution at timestep n ---*/
49 
50  if (TimeIter == 0) {
51  if (dual_time_2nd) {
52  /*--- Load solution at timestep n-2 ---*/
53 
54  LoadUnsteady_Solution(geometry, solver, config, val_iZone, val_iInst, Direct_Iter - 2);
55 
56  /*--- Push solution back to correct array ---*/
57 
58  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
59  solvers[iMesh][HEAT_SOL]->GetNodes()->Set_Solution_time_n();
60  solvers[iMesh][HEAT_SOL]->GetNodes()->Set_Solution_time_n1();
61  }
62  }
63  if (dual_time) {
64  /*--- Load solution at timestep n-1 ---*/
65 
66  LoadUnsteady_Solution(geometry, solver, config, val_iZone, val_iInst, Direct_Iter - 1);
67 
68  /*--- Push solution back to correct array ---*/
69 
70  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
71  solvers[iMesh][HEAT_SOL]->GetNodes()->Set_Solution_time_n();
72  }
73  }
74 
75  /*--- Load solution timestep n ---*/
76 
77  LoadUnsteady_Solution(geometry, solver, config, val_iZone, val_iInst, Direct_Iter);
78  }
79  if ((TimeIter > 0) && dual_time) {
80  /*--- Load solution timestep n - 2 ---*/
81 
82  LoadUnsteady_Solution(geometry, solver, config, val_iZone, val_iInst, Direct_Iter - 2);
83 
84  /*--- Temporarily store the loaded solution in the Solution_Old array ---*/
85 
86  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++)
87  solvers[iMesh][HEAT_SOL]->Set_OldSolution();
88 
89  /*--- Set Solution at timestep n to solution at n-1 ---*/
90 
91  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
92  for (auto iPoint = 0ul; iPoint < geometry[val_iZone][val_iInst][iMesh]->GetnPoint(); iPoint++) {
93  solvers[iMesh][HEAT_SOL]->GetNodes()->SetSolution(
94  iPoint, solvers[iMesh][HEAT_SOL]->GetNodes()->GetSolution_time_n(iPoint));
95  }
96  }
97  if (dual_time_1st) {
98  /*--- Set Solution at timestep n-1 to the previously loaded solution ---*/
99  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
100  for (auto iPoint = 0ul; iPoint < geometry[val_iZone][val_iInst][iMesh]->GetnPoint(); iPoint++) {
101  solvers[iMesh][HEAT_SOL]->GetNodes()->Set_Solution_time_n(
102  iPoint, solvers[iMesh][HEAT_SOL]->GetNodes()->GetSolution_time_n1(iPoint));
103  }
104  }
105  }
106  if (dual_time_2nd) {
107  /*--- Set Solution at timestep n-1 to solution at n-2 ---*/
108  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
109  for (auto iPoint = 0ul; iPoint < geometry[val_iZone][val_iInst][iMesh]->GetnPoint(); iPoint++) {
110  solvers[iMesh][HEAT_SOL]->GetNodes()->Set_Solution_time_n(
111  iPoint, solvers[iMesh][HEAT_SOL]->GetNodes()->GetSolution_time_n1(iPoint));
112  }
113  }
114  /*--- Set Solution at timestep n-2 to the previously loaded solution ---*/
115  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
116  for (auto iPoint = 0ul; iPoint < geometry[val_iZone][val_iInst][iMesh]->GetnPoint(); iPoint++) {
117  solvers[iMesh][HEAT_SOL]->GetNodes()->Set_Solution_time_n1(
118  iPoint, solvers[iMesh][HEAT_SOL]->GetNodes()->GetSolution_Old(iPoint));
119  }
120  }
121  }
122  }
123  }
124 
125  /*--- Store flow solution also in the adjoint solver in order to be able to reset it later ---*/
126 
127  if (TimeIter == 0 || dual_time) {
128  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
129  for (auto iPoint = 0ul; iPoint < geometry[val_iZone][val_iInst][iMesh]->GetnPoint(); iPoint++) {
130  solvers[iMesh][ADJHEAT_SOL]->GetNodes()->SetSolution_Direct(
131  iPoint, solvers[iMesh][HEAT_SOL]->GetNodes()->GetSolution(iPoint));
132  }
133  }
134  }
135 
136  solvers[MESH_0][ADJHEAT_SOL]->Preprocessing(
137  geometry[val_iZone][val_iInst][MESH_0], solvers[MESH_0], config[val_iZone], MESH_0, 0, RUNTIME_ADJHEAT_SYS, false);
138 }

140 void CDiscAdjHeatIteration::LoadUnsteady_Solution(CGeometry**** geometry, CSolver***** solver, CConfig** config,
141  unsigned short val_iZone, unsigned short val_iInst,
142  int val_DirectIter) {
143 
144  if (val_DirectIter >= 0) {
145  if (rank == MASTER_NODE)
146  cout << " Loading heat solution from direct iteration " << val_DirectIter << " for zone " << val_iZone << "." << endl;
147 
148  solver[val_iZone][val_iInst][MESH_0][HEAT_SOL]->LoadRestart(
149  geometry[val_iZone][val_iInst], solver[val_iZone][val_iInst], config[val_iZone], val_DirectIter, false);
150  }
151  else {
152  /*--- If there is no solution file we set the freestream condition ---*/
153  if (rank == MASTER_NODE)
154  cout << " Setting freestream conditions at direct iteration " << val_DirectIter << " for zone " << val_iZone << "." << endl;
155 
156  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
157  solver[val_iZone][val_iInst][iMesh][HEAT_SOL]->SetFreeStream_Solution(config[val_iZone]);
158  solver[val_iZone][val_iInst][iMesh][HEAT_SOL]->Postprocessing(
159  geometry[val_iZone][val_iInst][iMesh], solver[val_iZone][val_iInst][iMesh], config[val_iZone], iMesh);
160  }
161  }

178 void CDiscAdjHeatIteration::RegisterInput(CSolver***** solver, CGeometry**** geometry, CConfig** config,
179  unsigned short iZone, unsigned short iInst, RECORDING kind_recording) {
180 
181  if (kind_recording == RECORDING::SOLUTION_VARIABLES || kind_recording == RECORDING::SOLUTION_AND_MESH) {
182  /*--- Register flow and turbulent variables as input ---*/
183 
184  solver[iZone][iInst][MESH_0][ADJHEAT_SOL]->RegisterSolution(geometry[iZone][iInst][MESH_0], config[iZone]);
185 
186  solver[iZone][iInst][MESH_0][ADJHEAT_SOL]->RegisterVariables(geometry[iZone][iInst][MESH_0], config[iZone]);
187  }
188  else if (kind_recording == RECORDING::MESH_COORDS) {
189  /*--- Register node coordinates as input ---*/
190 
191  geometry[iZone][iInst][MESH_0]->RegisterCoordinates();
192  }
193  else if (kind_recording == RECORDING::MESH_DEFORM) {
194  /*--- Register the variables of the mesh deformation ---*/
195  /*--- Undeformed mesh coordinates ---*/
196  solver[iZone][iInst][MESH_0][ADJMESH_SOL]->RegisterSolution(geometry[iZone][iInst][MESH_0], config[iZone]);
197 
198  /*--- Boundary displacements ---*/
199  solver[iZone][iInst][MESH_0][ADJMESH_SOL]->RegisterVariables(geometry[iZone][iInst][MESH_0], config[iZone]);
200  }
201 }

203 void CDiscAdjHeatIteration::SetDependencies(CSolver***** solver, CGeometry**** geometry, CNumerics****** numerics,
204  CConfig** config, unsigned short iZone, unsigned short iInst,
205  RECORDING kind_recording) {
206 
207  auto solvers0 = solver[iZone][iInst][MESH_0];
208  auto geometries = geometry[iZone][iInst];
209 
210  if ((kind_recording == RECORDING::MESH_COORDS) ||
211  (kind_recording == RECORDING::CLEAR_INDICES) ||
212  (kind_recording == RECORDING::SOLUTION_AND_MESH)) {
213  /*--- Update geometry to get the influence on other geometry variables (normals, volume etc) ---*/
214 
215  CGeometry::UpdateGeometry(geometries, config[iZone]);
216 
217  CGeometry::ComputeWallDistance(config, geometry);
218  }
219 
220  solvers0[HEAT_SOL]->Set_Heatflux_Areas(geometries[MESH_0], config[iZone]);
221 
222  solvers0[HEAT_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone], MESH_0, NO_RK_ITER, RUNTIME_HEAT_SYS, true);
223  solvers0[HEAT_SOL]->Postprocessing(geometries[MESH_0], solvers0, config[iZone], MESH_0);
224 
225  solvers0[HEAT_SOL]->InitiateComms(geometries[MESH_0], config[iZone], SOLUTION);
226  solvers0[HEAT_SOL]->CompleteComms(geometries[MESH_0], config[iZone], SOLUTION);
227 }
