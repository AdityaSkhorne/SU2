205 void CHeatSolver::LoadRestart(CGeometry **geometry, CSolver ***solver, CConfig *config, int val_iter,
206  bool val_update_geo) {
207  /*--- Restart the solution from file information ---*/
208 
209  const string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
210 
211  /*--- Skip coordinates ---*/
212 
213  unsigned short skipVars = nDim;
214 
215  if (flow) {
216  // P, vx, vy (,vz)
217  if (nDim == 2) skipVars += 3;
218  if (nDim == 3) skipVars += 4;
219  }
220 
221  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
222 
223  if (config->GetRead_Binary_Restart()) {
224  Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
225  } else {
226  Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
227  }
228 
229  /*--- Load data from the restart into correct containers. ---*/
230  unsigned long counter = 0;
231  for (auto iPoint_Global = 0ul; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++ ) {
232 
233  /*--- Retrieve local index. If this node from the restart file lives
234  on the current processor, we will load and instantiate the vars. ---*/
235 
236  const auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
237 
238  if (iPoint_Local > -1) {
239  /*--- We need to store this point's data, so jump to the correct
240  offset in the buffer of data from the restart file and load it. ---*/
241 
242  const auto index = counter*Restart_Vars[1] + skipVars;
243  for (auto iVar = 0u; iVar < nVar; iVar++) nodes->SetSolution(iPoint_Local, iVar, Restart_Data[index + iVar]);
244 
245  /*--- Increment the overall counter for how many points have been loaded. ---*/
246  counter++;
247  }
248  }
249 
250  /*--- Detect a wrong solution file ---*/
251 
252  if (counter != nPointDomain) {
253  SU2_MPI::Error(string("The solution file ") + restart_filename + string(" does not match with the mesh file!\n") +
254  string("This can be caused by empty lines at the end of the file."), CURRENT_FUNCTION);
255  }
256 
257  /*--- Communicate the loaded solution on the fine grid before we transfer
258  it down to the coarse levels. We alo call the preprocessing routine
259  on the fine level in order to have all necessary quantities updated,
260  especially if this is a turbulent simulation (eddy viscosity). ---*/
261 
262  solver[MESH_0][HEAT_SOL]->InitiateComms(geometry[MESH_0], config, SOLUTION);
263  solver[MESH_0][HEAT_SOL]->CompleteComms(geometry[MESH_0], config, SOLUTION);
264 
265  solver[MESH_0][HEAT_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER, RUNTIME_HEAT_SYS, false);
266 
267  /*--- Interpolate the solution down to the coarse multigrid levels ---*/
268 
269  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
270  MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][HEAT_SOL]->GetNodes()->GetSolution(),
271  *geometry[iMesh], solver[iMesh][HEAT_SOL]->GetNodes()->GetSolution());
272  solver[iMesh][HEAT_SOL]->InitiateComms(geometry[iMesh], config, SOLUTION);
273  solver[iMesh][HEAT_SOL]->CompleteComms(geometry[iMesh], config, SOLUTION);
274 
275  solver[iMesh][HEAT_SOL]->Preprocessing(geometry[iMesh], solver[iMesh], config, iMesh, NO_RK_ITER, RUNTIME_HEAT_SYS, false);
276  }
277 
278  /*--- Delete the class memory that is used to load the restart. ---*/
279 
280  delete[] Restart_Vars;
281  Restart_Vars = nullptr;
282  delete[] Restart_Data;
283  Restart_Data = nullptr;
284 }

327 void CHeatSolver::Upwind_Residual(CGeometry *geometry, CSolver **solver_container,
328  CNumerics **numerics_container, CConfig *config, unsigned short iMesh) {
329 
330  CNumerics* numerics = numerics_container[CONV_TERM];
331 
332  su2double *V_i, *V_j, Temp_i, Temp_i_Corrected, Temp_j, Temp_j_Corrected,
333  Project_Grad_i, Project_Grad_j, Project_Temp_i_Grad, Project_Temp_j_Grad;
334 
335  su2double Vector_i[MAXNDIM] = {0};
336  su2double Vector_j[MAXNDIM] = {0};
337 
338  if(flow) {
339 
340  nVarFlow = solver_container[FLOW_SOL]->GetnVar();
341 
342  /*--- Define some auxiliary vectors related to the primitive flow solution ---*/
343  vector<su2double> Primitive_Flow_i(nVarFlow, 0.0);
344  vector<su2double> Primitive_Flow_j(nVarFlow, 0.0);
345 
346  for (auto iEdge = 0ul; iEdge < geometry->GetnEdge(); iEdge++) {
347 
348  /*--- Points in edge ---*/
349  const auto iPoint = geometry->edges->GetNode(iEdge,0);
350  const auto jPoint = geometry->edges->GetNode(iEdge,1);
351  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
352 
353  /*--- Primitive variables w/o reconstruction ---*/
354  V_i = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
355  V_j = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(jPoint);
356 
357  const auto Temp_i_Grad = nodes->GetGradient(iPoint);
358  const auto Temp_j_Grad = nodes->GetGradient(jPoint);
359  numerics->SetConsVarGradient(Temp_i_Grad, Temp_j_Grad);
360 
361  Temp_i = nodes->GetTemperature(iPoint);
362  Temp_j = nodes->GetTemperature(jPoint);
363 
364  /* Second order reconstruction */
365  if (config->GetMUSCL_Heat()) {
366 
367  for (auto iDim = 0u; iDim < nDim; iDim++) {
368  Vector_i[iDim] = 0.5*(geometry->nodes->GetCoord(jPoint, iDim) - geometry->nodes->GetCoord(iPoint, iDim));
369  Vector_j[iDim] = 0.5*(geometry->nodes->GetCoord(iPoint, iDim) - geometry->nodes->GetCoord(jPoint, iDim));
370  }
371 
372  const auto Gradient_i = solver_container[FLOW_SOL]->GetNodes()->GetGradient_Reconstruction(iPoint);
373  const auto Gradient_j = solver_container[FLOW_SOL]->GetNodes()->GetGradient_Reconstruction(jPoint);
374  const auto Temp_i_Grad = nodes->GetGradient_Reconstruction(iPoint);
375  const auto Temp_j_Grad = nodes->GetGradient_Reconstruction(jPoint);
376 
377  /*Loop to correct the flow variables*/
378  for (auto iVar = 0u; iVar < nVarFlow; iVar++) {
379 
380  /*Apply the Gradient to get the right temperature value on the edge */
381  Project_Grad_i = 0.0; Project_Grad_j = 0.0;
382  for (auto iDim = 0u; iDim < nDim; iDim++) {
383  Project_Grad_i += Vector_i[iDim]*Gradient_i[iVar][iDim];
384  Project_Grad_j += Vector_j[iDim]*Gradient_j[iVar][iDim];
385  }
386 
387  Primitive_Flow_i[iVar] = V_i[iVar] + Project_Grad_i;
388  Primitive_Flow_j[iVar] = V_j[iVar] + Project_Grad_j;
389  }
390 
391  /* Correct the temperature variables */
392  Project_Temp_i_Grad = 0.0; Project_Temp_j_Grad = 0.0;
393  for (auto iDim = 0u; iDim < nDim; iDim++) {
394  Project_Temp_i_Grad += Vector_i[iDim]*Temp_i_Grad[0][iDim];
395  Project_Temp_j_Grad += Vector_j[iDim]*Temp_j_Grad[0][iDim];
396  }
397 
398  Temp_i_Corrected = Temp_i + Project_Temp_i_Grad;
399  Temp_j_Corrected = Temp_j + Project_Temp_j_Grad;
400 
401  numerics->SetPrimitive(Primitive_Flow_i.data(), Primitive_Flow_j.data());
402  numerics->SetTemperature(Temp_i_Corrected, Temp_j_Corrected);
403  }
404 
405  else {
406 
407  numerics->SetPrimitive(V_i, V_j);
408  numerics->SetTemperature(Temp_i, Temp_j);
409  }
410 
411  numerics->ComputeResidual(Residual, Jacobian_i, Jacobian_j, config);
412 
413  LinSysRes.AddBlock(iPoint, Residual);
414  LinSysRes.SubtractBlock(jPoint, Residual);
415 
416  /*--- Implicit part ---*/
417 
418  Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, Jacobian_i, Jacobian_j);
419  }
420  }
421

424 void CHeatSolver::Viscous_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
425  CConfig *config, unsigned short iMesh, unsigned short iRKStep) {
426 
427  CNumerics* numerics = numerics_container[VISC_TERM];
428 
429  const CVariable* flow_nodes = flow ? solver_container[FLOW_SOL]->GetNodes() : nullptr;
430 
431  const su2double const_diffusivity = config->GetThermalDiffusivity();
432  const su2double pr_lam = config->GetPrandtl_Lam();
433  const su2double pr_turb = config->GetPrandtl_Turb();
434 
435  for (auto iEdge = 0ul; iEdge < geometry->GetnEdge(); iEdge++) {
436  su2double thermal_diffusivity_i{}, thermal_diffusivity_j{};
437 
438  /*--- Computes the thermal diffusivity to use in the viscous numerics. ---*/
439  auto compute_thermal_diffusivity = [&](unsigned long iPoint, unsigned long jPoint) {
440  if (flow) {
441  thermal_diffusivity_i = flow_nodes->GetLaminarViscosity(iPoint) / pr_lam +
442  flow_nodes->GetEddyViscosity(iPoint) / pr_turb;
443  thermal_diffusivity_j = flow_nodes->GetLaminarViscosity(jPoint) / pr_lam +
444  flow_nodes->GetEddyViscosity(jPoint) / pr_turb;
445  numerics->SetDiffusionCoeff(&thermal_diffusivity_i, &thermal_diffusivity_j);
446  }
447  else {
448  numerics->SetDiffusionCoeff(&const_diffusivity, &const_diffusivity);
449  }
450  };
451  /*--- Compute residual and Jacobians. ---*/
452  Viscous_Residual_impl(compute_thermal_diffusivity, iEdge, geometry, solver_container, numerics, config);
453  }

522 void CHeatSolver::BC_HeatFlux_Wall(CGeometry* geometry, CSolver** solver_container, CNumerics* conv_numerics,
523  CNumerics* visc_numerics, CConfig* config, unsigned short val_marker) {
524  const auto Marker_Tag = config->GetMarker_All_TagBound(val_marker);
525 
526  su2double Wall_HeatFlux = config->GetWall_HeatFlux(Marker_Tag) / config->GetHeat_Flux_Ref();
527  if (config->GetIntegrated_HeatFlux()) {
528  Wall_HeatFlux /= geometry->GetSurfaceArea(config, val_marker);
529  }
530 
531  for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
532  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
533  if (!geometry->nodes->GetDomain(iPoint)) continue;

862 void CHeatSolver::SetTime_Step(CGeometry *geometry, CSolver **solver_container, CConfig *config,
863  unsigned short iMesh, unsigned long Iteration) {
864 
865  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
866  const bool time_stepping = (config->GetTime_Marching() == TIME_MARCHING::TIME_STEPPING);
867  const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
868  (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
869  const su2double K_v = 0.25;
870 
871  const su2double prandtl_lam = config->GetPrandtl_Lam();
872  const su2double prandtl_turb = config->GetPrandtl_Turb();
873  const su2double constant_thermal_diffusivity = config->GetThermalDiffusivity();
874 
875  const CVariable* flow_nodes = solver_container[FLOW_SOL] ? solver_container[FLOW_SOL]->GetNodes() : nullptr;
876 
877  /*--- Init thread-shared variables to compute min/max values.
878  * Critical sections are used for this instead of reduction
879  * clauses for compatibility with OpenMP 2.0 (Windows...). ---*/
880  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
881  Min_Delta_Time = 1e30;
882  Max_Delta_Time = 0.0;
883  Global_Delta_UnstTimeND = 1e30;
884  } END_SU2_OMP_SAFE_GLOBAL_ACCESS
885 
886  /*--- Loop domain points. ---*/
887 
888  SU2_OMP_FOR_DYN(omp_chunk_size)
889  for (auto iPoint = 0ul; iPoint < nPointDomain; ++iPoint) {
890 
891  /*--- Set maximum eigenvalues to zero. ---*/
892 
893  nodes->SetMax_Lambda_Visc(iPoint, 0.0);
894 
895  /*--- Loop over the neighbors of point i. ---*/
896 
897  for (unsigned short iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); ++iNeigh) {
898  const auto jPoint = geometry->nodes->GetPoint(iPoint,iNeigh);
899  const auto iEdge = geometry->nodes->GetEdge(iPoint,iNeigh);
900  const auto* Normal = geometry->edges->GetNormal(iEdge);
901  const su2double Area2 = GeometryToolbox::SquaredNorm(nDim, Normal);
902 
903  if (flow) {
904  const su2double laminar_viscosity = 0.5 * (flow_nodes->GetLaminarViscosity(iPoint) +
905  flow_nodes->GetLaminarViscosity(jPoint));
906  const su2double eddy_viscosity = 0.5 * (flow_nodes->GetEddyViscosity(iPoint) +
907  flow_nodes->GetEddyViscosity(jPoint));
908 
909  const su2double thermal_diffusivity = laminar_viscosity / prandtl_lam + eddy_viscosity / prandtl_turb;
910  nodes->AddMax_Lambda_Visc(iPoint, thermal_diffusivity * Area2);
911  } else {
912  nodes->AddMax_Lambda_Visc(iPoint, constant_thermal_diffusivity * Area2);
913  }
914  }
915  }
916  END_SU2_OMP_FOR
917 
918  /*--- Loop boundary edges ---*/
919 
920  for (unsigned short iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
921  if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
922  (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
923  (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
924 
925  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
926  for (auto iVertex = 0ul; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
927 
928  /*--- Point identification, Normal vector and area ---*/
929 
930  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
931 
932  if (!geometry->nodes->GetDomain(iPoint)) continue;
933 
934  const auto* Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
935  const su2double Area2 = GeometryToolbox::SquaredNorm(nDim, Normal);
936 
937  if (flow) {
938  const su2double thermal_diffusivity = flow_nodes->GetLaminarViscosity(iPoint) / prandtl_lam +
939  flow_nodes->GetEddyViscosity(iPoint) / prandtl_turb;
940  nodes->AddMax_Lambda_Visc(iPoint, thermal_diffusivity * Area2);
941  } else {
942  nodes->AddMax_Lambda_Visc(iPoint, constant_thermal_diffusivity * Area2);
943  }
944  }
945  END_SU2_OMP_FOR
946  }
947  }
948 
949  /*--- Each element uses their own speed, steady state simulation. ---*/
950  {
951  /*--- Thread-local variables for min/max reduction. ---*/
952  su2double minDt = 1e30, maxDt = 0.0;
953 
954  SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
955  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
956 
957  su2double local_delta_time = 0.0;
958 
959  const su2double Vol = geometry->nodes->GetVolume(iPoint);
960  if (Vol != 0.0) {
961  local_delta_time = nodes->GetLocalCFL(iPoint) * K_v * pow(Vol, 2) / nodes->GetMax_Lambda_Visc(iPoint);
962  if (flow) {
963  switch (config->GetKind_TimeStep_Heat()) {
964  case BYFLOW:
965  case CONVECTIVE:
966  local_delta_time = flow_nodes->GetDelta_Time(iPoint);
967  break;
968  case MINIMUM:
969  local_delta_time = min(local_delta_time, flow_nodes->GetDelta_Time(iPoint));
970  break;
971  case VISCOUS:
972  break;
973  default:
974  break;
975  }
976  }
977  minDt = min(minDt, local_delta_time);
978  maxDt = max(maxDt, local_delta_time);
979  }
980  nodes->SetDelta_Time(iPoint, local_delta_time);
981  }
982  END_SU2_OMP_FOR
983  /*--- Min/max over threads. ---*/
984  SU2_OMP_CRITICAL
985  {
986  Min_Delta_Time = min(Min_Delta_Time, minDt);
987  Max_Delta_Time = max(Max_Delta_Time, maxDt);
988  Global_Delta_Time = Min_Delta_Time;
989  }
990  END_SU2_OMP_CRITICAL
991  }
992 
993  /*--- Compute the min/max dt (in parallel, now over mpi ranks). ---*/
994 
995  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
996  if (config->GetComm_Level() == COMM_FULL) {
997  su2double rbuf_time;
998  SU2_MPI::Allreduce(&Min_Delta_Time, &rbuf_time, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
999  Min_Delta_Time = rbuf_time;
1000 
1001  SU2_MPI::Allreduce(&Max_Delta_Time, &rbuf_time, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
1002  Max_Delta_Time = rbuf_time;
1003  }
1004  } END_SU2_OMP_SAFE_GLOBAL_ACCESS
1005 
1006  /*--- For exact time solution use the minimum delta time of the whole mesh. ---*/
1007  if (time_stepping) {
1008 
1009  /*--- If the unsteady CFL is set to zero, it uses the defined unsteady time step,
1010  * otherwise it computes the time step based on the unsteady CFL. ---*/
1011 
1012  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
1013  {
1014  if (config->GetUnst_CFL() == 0.0) {
1015  Global_Delta_Time = config->GetDelta_UnstTime();
1016  }
1017  else {
1018  Global_Delta_Time = Min_Delta_Time;
1019  }
1020  Max_Delta_Time = Global_Delta_Time;
1021 
1022  config->SetDelta_UnstTimeND(Global_Delta_Time);
1023  }
1024  END_SU2_OMP_SAFE_GLOBAL_ACCESS
1025 
1026  /*--- Sets the regular CFL equal to the unsteady CFL. ---*/
1027 
1028  SU2_OMP_FOR_STAT(omp_chunk_size)
1029  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
1030  nodes->SetLocalCFL(iPoint, config->GetUnst_CFL());
1031  nodes->SetDelta_Time(iPoint, Global_Delta_Time);
1032  }
1033  END_SU2_OMP_FOR
1034 
1035  }
1036 
1037  /*--- Recompute the unsteady time step for the dual time strategy if the unsteady CFL is diferent from 0.
1038  * This is only done once because in dual time the time step cannot be variable. ---*/
1039 
1040  if (dual_time && (Iteration == config->GetRestart_Iter()) && (config->GetUnst_CFL() != 0.0) && (iMesh == MESH_0)) {
1041 
1042  /*--- Thread-local variable for reduction. ---*/
1043  su2double glbDtND = 1e30;
1044 
1045  SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
1046  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
1047  glbDtND = min(glbDtND, config->GetUnst_CFL()*Global_Delta_Time / nodes->GetLocalCFL(iPoint));
1048  }
1049  END_SU2_OMP_FOR
1050  SU2_OMP_CRITICAL
1051  Global_Delta_UnstTimeND = min(Global_Delta_UnstTimeND, glbDtND);
1052  END_SU2_OMP_CRITICAL
1053 
1054  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
1055  {
1056  SU2_MPI::Allreduce(&Global_Delta_UnstTimeND, &glbDtND, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
1057  Global_Delta_UnstTimeND = glbDtND;
1058 
1059  config->SetDelta_UnstTimeND(Global_Delta_UnstTimeND);
1060  }
1061  END_SU2_OMP_SAFE_GLOBAL_ACCESS
1062  }
1063 
1064  /*--- The pseudo local time (explicit integration) cannot be greater than the physical time ---*/
1065 
1066  if (dual_time && !implicit) {
1067  SU2_OMP_FOR_STAT(omp_chunk_size)
1068  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
1069  su2double dt = min((2.0/3.0)*config->GetDelta_UnstTimeND(), nodes->GetDelta_Time(iPoint));
1070  nodes->SetDelta_Time(iPoint, dt);
1071  }
1072  END_SU2_OMP_FOR
1073  }
1074 }
